49template<class KeyType>
50inline HOST_DEVICE_FUN int mergeCountAndMacOp(TreeNodeIndex nodeIdx,
51 const KeyType* nodeKeys,
52 const TreeNodeIndex* childOffsets,
53 const TreeNodeIndex* parents,
54 const unsigned* counts,
55 const char* macs,
56 KeyType focusStart,
57 KeyType focusEnd,
58 unsigned bucketSize)
59{
60 TreeNodeIndex siblingGroup = (nodeIdx - 1) / 8;
61 TreeNodeIndex parent = nodeIdx ? parents[siblingGroup] : 0;
62 KeyType nodeKey = nodeKeys[nodeIdx];
63 unsigned level = decodePrefixLength(nodeKeys[nodeIdx]) / 3;
64
65 if (nodeIdx)
66 {
67 bool countMerge = counts[parent] <= bucketSize;
68 bool macMerge = macs[parent] == 0;
69
70 KeyType firstGroupKey = decodePlaceholderBit(nodeKeys[parent]);
71 KeyType lastGroupKey = firstGroupKey + 8 * nodeRange<KeyType>(level);
72 // inFringe: nodeIdx not in focus, but at least one sibling is in the focus
73 // in that case we cannot remove the nodes based on a MAC criterion
74 bool inFringe = overlapTwoRanges(firstGroupKey, lastGroupKey, focusStart, focusEnd);
75
76 if (countMerge || (macMerge && !inFringe)) { return 0; } // merge
77 }
78
79 KeyType nodeStart = decodePlaceholderBit(nodeKey);
80 bool isLeaf = childOffsets[nodeIdx] == 0;
81 bool inFocus = (nodeStart >= focusStart && nodeStart < focusEnd);
82 if (isLeaf && level < maxTreeLevel<KeyType>{} && counts[nodeIdx] > bucketSize && (macs[nodeIdx] || inFocus))
83 {
84 return 8; // split
85 }
86
87 return 1; // default: do nothing
88}

102template<class KeyType>
103inline HOST_DEVICE_FUN TreeNodeIndex
104nzAncestorOp(TreeNodeIndex nodeIdx, const KeyType* prefixes, const TreeNodeIndex* parents, const TreeNodeIndex* nodeOps)
105{
106 if (nodeIdx == 0) { return nodeOps[0]; }
107
108 TreeNodeIndex closestNonZeroAncestor = nodeIdx;
109 while (nodeOps[closestNonZeroAncestor] == 0)
110 {
111 closestNonZeroAncestor = parents[(closestNonZeroAncestor - 1) / 8];
112 }
113
114 if (decodePlaceholderBit(prefixes[nodeIdx]) == decodePlaceholderBit(prefixes[closestNonZeroAncestor]))
115 {
116 return nodeOps[closestNonZeroAncestor];
117 }
118
119 return 0;
120}

143template<class KeyType>
144void rebalanceDecisionEssential(gsl::span<const KeyType> nodeKeys,
145 const TreeNodeIndex* childOffsets,
146 const TreeNodeIndex* parents,

162template<class KeyType>
163bool protectAncestors(gsl::span<const KeyType> nodeKeys, const TreeNodeIndex* parents, TreeNodeIndex* nodeOps)
164{
165 int numChanges = 0;
166#pragma omp parallel for reduction(+ : numChanges)
167 for (TreeNodeIndex i = 0; i < nodeKeys.ssize(); ++i)
168 {
169 int opDecision = nzAncestorOp(i, nodeKeys.data(), parents, nodeOps);
170
171 if (opDecision != 1) { numChanges++; }
172 nodeOps[i] = opDecision;
173 }
174 return numChanges == 0;
175}

189template<class KeyType>
190HOST_DEVICE_FUN ResolutionStatus enforceKeySingle(KeyType key,
191 const KeyType* nodeKeys,
192 const TreeNodeIndex* childOffsets,
193 const TreeNodeIndex* parents,
194 TreeNodeIndex* nodeOps)
195{
196 if (key == 0 || key == nodeRange<KeyType>(0)) { return ResolutionStatus::converged; }
197
198 ResolutionStatus status = ResolutionStatus::converged;
199
200 KeyType nodeKeyWant = makePrefix(key);
201 TreeNodeIndex nodeIdx = containingNode(nodeKeyWant, nodeKeys, childOffsets);
202 KeyType nodeKeyHave = nodeKeys[nodeIdx];
203 unsigned nodeLevelHave = decodePrefixLength(nodeKeyHave) / 3;
204
205 // need to undo merges of all supporting ancestors if the closest tree node would be merged
206 // or the mandatory key is not there
207 bool trySplit = nodeKeyHave != nodeKeyWant && nodeLevelHave < maxTreeLevel<KeyType>{};
208 bool undoMerges = nodeOps[nodeIdx] == 0 || trySplit;
209 if (undoMerges && nodeIdx > 0)
210 {
211 status = ResolutionStatus::cancelMerge;
212 // make sure no supporting ancestor of nodeIdx will be merged
213 TreeNodeIndex parent = nodeIdx;
214 do
215 {
216 parent = parents[(parent - 1) / 8];
217 TreeNodeIndex firstSibling = childOffsets[parent];
218 for (TreeNodeIndex i = firstSibling; i < firstSibling + eightSiblings; ++i)
219 {
220 if (nodeOps[i] == 0) { nodeOps[i] = 1; }
221 }
222 } while (parent != 0);
223 }
224
225 if (trySplit)
226 {
227 int keyPos = lastNzPlace(key);
228
229 // only add 1 level, otherwise nodes can be added in a non-peer area,
230 // exceeding the resolution of the global tree, which will result in a failure to compute
231 // exact particle counts for those nodes
232 constexpr int maxAddLevels = 1;
233 int levelDiff = keyPos - nodeLevelHave;
234 if (levelDiff > maxAddLevels) { status = ResolutionStatus::failed; }
235 else { status = ResolutionStatus::rebalance; }
236
237 levelDiff = stl::min(levelDiff, maxAddLevels);
238 nodeOps[nodeIdx] = stl::max(nodeOps[nodeIdx], 1 << (3 * levelDiff));
239 }
240
241 return status;
242}

244template<class KeyType>
245ResolutionStatus enforceKeys(gsl::span<const KeyType> mandatoryKeys,
246 const KeyType* nodeKeys,
247 const TreeNodeIndex* childOffsets,
248 const TreeNodeIndex* parents,
249 TreeNodeIndex* nodeOps)
250{
251 ResolutionStatus status = ResolutionStatus::converged;
252
253 for (KeyType key : mandatoryKeys)
254 {
255 status = std::max(enforceKeySingle(key, nodeKeys, childOffsets, parents, nodeOps), status);
256 }
257 return status;
258}

270template<class KeyType, class Alloc>
271void injectKeys(std::vector<KeyType, Alloc>& tree, gsl::span<const KeyType> keys)
272{
273 std::vector<KeyType> spanningKeys(keys.begin(), keys.end());
274 spanningKeys.push_back(0);
275 spanningKeys.push_back(nodeRange<KeyType>(0));
276 std::sort(begin(spanningKeys), end(spanningKeys));
277 auto uit = std::unique(begin(spanningKeys), end(spanningKeys));
278 spanningKeys.erase(uit, end(spanningKeys));
279
280 // spanningTree is a list of all the missing nodes needed to resolve the mandatory keys
281 auto spanningTree = computeSpanningTree<KeyType>(spanningKeys);
282 tree.reserve(tree.size() + spanningTree.size());
283
284 // spanningTree is now inserted into newLeaves
285 std::copy(begin(spanningTree), end(spanningTree), std::back_inserter(tree));
286
287 // cleanup, restore invariants: sorted-ness, no-duplicates
288 std::sort(begin(tree), end(tree));
289 uit = std::unique(begin(tree), end(tree));
290 tree.erase(uit, end(tree));
291}
