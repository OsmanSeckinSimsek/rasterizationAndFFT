71template<class KeyType>
72HOST_DEVICE_FUN constexpr TreeNodeIndex binaryKeyWeight(KeyType key, unsigned level)
73{
74 TreeNodeIndex ret = 0;
75 for (unsigned l = 1; l <= level + 1; ++l)
76 {
77 unsigned digit = octalDigit(key, l);
78 ret += digitWeight(digit);
79 }
80 return ret;

94template<class KeyType>
95void createUnsortedLayoutCpu(const KeyType* leaves,
96 TreeNodeIndex numInternalNodes,
97 TreeNodeIndex numLeafNodes,
98 KeyType* prefixes,
99 TreeNodeIndex* internalToLeaf)
100{
101#pragma omp parallel for schedule(static)
102 for (TreeNodeIndex tid = 0; tid < numLeafNodes; ++tid)
103 {
104 KeyType key = leaves[tid];
105 unsigned level = treeLevel(leaves[tid + 1] - key);
106 prefixes[tid + numInternalNodes] = encodePlaceholderBit(key, 3 * level);
107 internalToLeaf[tid + numInternalNodes] = tid + numInternalNodes;
108
109 unsigned prefixLength = commonPrefix(key, leaves[tid + 1]);
110 if (prefixLength % 3 == 0 && tid < numLeafNodes - 1)
111 {
112 TreeNodeIndex octIndex = (tid + binaryKeyWeight(key, prefixLength / 3)) / 7;
113 prefixes[octIndex] = encodePlaceholderBit(key, prefixLength);
114 internalToLeaf[octIndex] = octIndex;
115 }
116 }
117}

131template<class KeyType>
132void linkTreeCpu(const KeyType* prefixes,
133 TreeNodeIndex numInternalNodes,
134 const TreeNodeIndex* leafToInternal,
135 const TreeNodeIndex* levelRange,
136 TreeNodeIndex* childOffsets,
137 TreeNodeIndex* parents)
138{
139#pragma omp parallel for schedule(static)
140 for (TreeNodeIndex i = 0; i < numInternalNodes; ++i)
141 {
142 TreeNodeIndex idxA = leafToInternal[i];
143 KeyType prefix = prefixes[idxA];
144 KeyType nodeKey = decodePlaceholderBit(prefix);
145 unsigned prefixLength = decodePrefixLength(prefix);
146 unsigned level = prefixLength / 3;
147 assert(level < maxTreeLevel<KeyType>{});
148
149 KeyType childPrefix = encodePlaceholderBit(nodeKey, prefixLength + 3);
150
151 TreeNodeIndex leafSearchStart = levelRange[level + 1];
152 TreeNodeIndex leafSearchEnd = levelRange[level + 2];
153 TreeNodeIndex childIdx =
154 stl::lower_bound(prefixes + leafSearchStart, prefixes + leafSearchEnd, childPrefix) - prefixes;
155
156 if (childIdx != leafSearchEnd && childPrefix == prefixes[childIdx])
157 {
158 childOffsets[idxA] = childIdx;
159 // We only store the parent once for every group of 8 siblings.
160 // This works as long as each node always has 8 siblings.
161 // Subtract one because the root has no siblings.
162 parents[(childIdx - 1) / 8] = idxA;
163 }
164 }
165}

168template<class KeyType>
169void getLevelRangeCpu(const KeyType* nodeKeys, TreeNodeIndex numNodes, TreeNodeIndex* levelRange)
170{
171 for (unsigned level = 0; level <= maxTreeLevel<KeyType>{}; ++level)
172 {
173 auto it = std::lower_bound(nodeKeys, nodeKeys + numNodes, encodePlaceholderBit(KeyType(0), 3 * level));
174 levelRange[level] = TreeNodeIndex(it - nodeKeys);
175 }
176 levelRange[maxTreeLevel<KeyType>{} + 1] = numNodes;
177}

184template<class KeyType>
185void buildOctreeCpu(const KeyType* cstoneTree,
186 TreeNodeIndex numLeafNodes,
187 TreeNodeIndex numInternalNodes,
188 KeyType* prefixes,
189 TreeNodeIndex* childOffsets,
190 TreeNodeIndex* parents,
191 TreeNodeIndex* levelRange,
192 TreeNodeIndex* internalToLeaf,
193 TreeNodeIndex* leafToInternal)
194{
195 TreeNodeIndex numNodes = numInternalNodes + numLeafNodes;
196 createUnsortedLayoutCpu(cstoneTree, numInternalNodes, numLeafNodes, prefixes, internalToLeaf);
197 sort_by_key(prefixes, prefixes + numNodes, internalToLeaf);
198
199#pragma omp parallel for schedule(static)
200 for (TreeNodeIndex i = 0; i < numNodes; ++i)
201 {
202 leafToInternal[internalToLeaf[i]] = i;
203 }
204#pragma omp parallel for schedule(static)
205 for (TreeNodeIndex i = 0; i < numNodes; ++i)
206 {
207 internalToLeaf[i] -= numInternalNodes;
208 }
209 getLevelRangeCpu(prefixes, numNodes, levelRange);
210
211 std::fill(childOffsets, childOffsets + numNodes, 0);
212 linkTreeCpu(prefixes, numInternalNodes, leafToInternal, levelRange, childOffsets, parents);
213}

216template<class KeyType>
217HOST_DEVICE_FUN TreeNodeIndex locateNode(KeyType nodeKey, const KeyType* prefixes, const TreeNodeIndex* levelRange)
218{
219 TreeNodeIndex numNodes = levelRange[maxTreeLevel<KeyType>{} + 1];
220 unsigned level = decodePrefixLength(nodeKey) / 3;
221 auto it = stl::lower_bound(prefixes + levelRange[level], prefixes + levelRange[level + 1], nodeKey);
222 if (it != prefixes + numNodes && *it == nodeKey) { return it - prefixes; }
223 else { return numNodes; }
224}

233template<class KeyType>
234HOST_DEVICE_FUN TreeNodeIndex
235locateNode(KeyType startKey, KeyType endKey, const KeyType* prefixes, const TreeNodeIndex* levelRange)
236{
238 unsigned prefixLength = countLeadingZeros(endKey - startKey - 1) - unusedBits<KeyType>{};
239 return locateNode(encodePlaceholderBit(startKey, prefixLength), prefixes, levelRange);
240}

243template<class KeyType>
244HOST_DEVICE_FUN TreeNodeIndex containingNode(KeyType nodeKey,
245 const KeyType* prefixes,
246 const TreeNodeIndex* childOffsets)
247{
248 int nodeLevel = decodePrefixLength(nodeKey) / 3;
249 KeyType key = decodePlaceholderBit(nodeKey);
250
251 TreeNodeIndex ret = 0;
252 for (int i = 1; i <= nodeLevel; ++i)
253 {
254 if (childOffsets[ret] == 0 || nodeKey == prefixes[ret]) { break; }
255
256 ret = childOffsets[ret] + octalDigit(key, i);
257 }
258
259 return ret;
260}

601template<class T, class CombinationFunction>
602void upsweep(gsl::span<const TreeNodeIndex> levelOffset,
603 gsl::span<const TreeNodeIndex> childOffsets,
604 T* quantities,
605 CombinationFunction&& combinationFunction)
606{
607 int currentLevel = levelOffset.size() - 2;
608
609 for (; currentLevel >= 0; --currentLevel)
610 {
611 TreeNodeIndex start = levelOffset[currentLevel];
612 TreeNodeIndex end = levelOffset[currentLevel + 1];
613#pragma omp parallel for schedule(static)
614 for (TreeNodeIndex i = start; i < end; ++i)
615 {
616 cstone::TreeNodeIndex firstChild = childOffsets[i];
617 if (firstChild) { quantities[i] = combinationFunction(i, firstChild, quantities); }
618 }
619 }
620}

631template<class CountType>
632struct NodeCount
633{
634 CountType operator()(TreeNodeIndex /*nodeIdx*/, TreeNodeIndex c, const CountType* Q)
635 {
636 uint64_t sum = Q[c];
637 for (TreeNodeIndex octant = 1; octant < 8; ++octant)
638 {
639 sum += Q[c + octant];
640 }
641 return stl::min(uint64_t(std::numeric_limits<CountType>::max()), sum);
