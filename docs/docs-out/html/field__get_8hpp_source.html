41template<class Dataset, class Tuple, util::StructuralString... Fields>
42decltype(auto) getFields(Tuple&& tuple, util::FieldList<Fields...>)
43{
44 if constexpr (sizeof...(Fields) == 1)

75template<class FL, class Dataset>
76decltype(auto) get(Dataset& d)
77{
78 using AcceleratorType = typename Dataset::AcceleratorType;
79 if constexpr (std::is_same_v<AcceleratorType, CpuTag>) { return getHost<FL>(d); }
