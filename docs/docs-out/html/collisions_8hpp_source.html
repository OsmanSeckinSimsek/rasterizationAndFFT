39template<class KeyType, class F>
40HOST_DEVICE_FUN void findCollisions(const KeyType* nodePrefixes,
41 const TreeNodeIndex* childOffsets,
42 F&& endpointAction,
43 const IBox& target,
44 KeyType excludeStart,
45 KeyType excludeEnd)
46{
47 auto overlaps = [excludeStart, excludeEnd, nodePrefixes, &target](TreeNodeIndex idx)
48 {
49 KeyType nodeKey = decodePlaceholderBit(nodePrefixes[idx]);
50 int level = decodePrefixLength(nodePrefixes[idx]) / 3;
51 IBox sourceBox = sfcIBox(sfcKey(nodeKey), level);
52 return !containedIn(nodeKey, nodeKey + nodeRange<KeyType>(level), excludeStart, excludeEnd) &&
53 overlap<KeyType>(sourceBox, target);
54 };
55
56 singleTraversal(childOffsets, overlaps, endpointAction);
57}

78template<class KeyType, class RadiusType, class CoordinateType>
79void findHalos(const KeyType* prefixes,
80 const TreeNodeIndex* childOffsets,
81 const TreeNodeIndex* internalToLeaf,
82 const KeyType* leaves,
83 const RadiusType* interactionRadii,
84 const Box<CoordinateType>& box,
85 TreeNodeIndex firstNode,
86 TreeNodeIndex lastNode,
87 int* collisionFlags)
88{
89 KeyType lowestCode = leaves[firstNode];
90 KeyType highestCode = leaves[lastNode];
91
92 auto markCollisions = [collisionFlags, internalToLeaf](TreeNodeIndex i) { collisionFlags[internalToLeaf[i]] = 1; };
93
94#pragma omp parallel for
95 for (TreeNodeIndex nodeIdx = firstNode; nodeIdx < lastNode; ++nodeIdx)
96 {
97 RadiusType radius = interactionRadii[nodeIdx];
98 IBox haloBox = makeHaloBox<KeyType>(leaves[nodeIdx], leaves[nodeIdx + 1], radius, box);
99
100 // if the halo box is fully inside the assigned SFC range, we skip collision detection
101 if (containedIn(lowestCode, highestCode, haloBox)) { continue; }
102
103 findCollisions(prefixes, childOffsets, markCollisions, haloBox, lowestCode, highestCode);
104 }
105}
