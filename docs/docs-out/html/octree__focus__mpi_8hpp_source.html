109 bool updateTree(gsl::span<const int> peerRanks,
110 const SpaceCurveAssignment& assignment,
111 gsl::span<const KeyType> globalTreeLeaves)
112 {
113 if (rebalanceStatus_ != valid)
114 {
115 throw std::runtime_error("update of criteria required before updating the tree structure\n");
116 }
117 peers_.resize(peerRanks.size());
118 std::copy(peerRanks.begin(), peerRanks.end(), peers_.begin());
119
120 KeyType focusStart = globalTreeLeaves[assignment.firstNodeIdx(myRank_)];
121 KeyType focusEnd = globalTreeLeaves[assignment.lastNodeIdx(myRank_)];
122 // init on first call
123 if (prevFocusStart == 0 && prevFocusEnd == 0)
124 {
125 prevFocusStart = focusStart;
126 prevFocusEnd = focusEnd;
127 }
128
129 std::vector<KeyType> enforcedKeys;
130 enforcedKeys.reserve(peers_.size() * 2);
131
132 focusTransfer<KeyType>(leaves_, leafCounts_, bucketSize_, myRank_, prevFocusStart, prevFocusEnd, focusStart,
133 focusEnd, enforcedKeys);
134 for (int peer : peers_)
135 {
136 enforcedKeys.push_back(globalTreeLeaves[assignment.firstNodeIdx(peer)]);
137 enforcedKeys.push_back(globalTreeLeaves[assignment.lastNodeIdx(peer)]);
138 }
139 auto uniqueEnd = std::unique(enforcedKeys.begin(), enforcedKeys.end());
140 enforcedKeys.erase(uniqueEnd, enforcedKeys.end());
141
142 bool converged;
143 if constexpr (HaveGpu<Accelerator>{})
144 {
145 converged = CombinedUpdate<KeyType>::updateFocusGpu(
146 octreeAcc_, leavesAcc_, bucketSize_, focusStart, focusEnd, enforcedKeys,
147 {rawPtr(countsAcc_), countsAcc_.size()}, {rawPtr(macsAcc_), macsAcc_.size()});
148 }
149 else
150 {
151 converged = CombinedUpdate<KeyType>::updateFocus(treeData_, leaves_, bucketSize_, focusStart, focusEnd,
152 enforcedKeys, counts_, macs_);
153 }
154 downloadOctree();
155
156 translateAssignment<KeyType>(assignment, globalTreeLeaves, leaves_, peers_, myRank_, assignment_);
157
158 prevFocusStart = focusStart;
159 prevFocusEnd = focusEnd;
160 rebalanceStatus_ = invalid;
161 return converged;
162 }

179 template<class DeviceVector = std::vector<KeyType>>
180 void updateCounts(gsl::span<const KeyType> particleKeys,
181 gsl::span<const KeyType> globalTreeLeaves,
182 gsl::span<const unsigned> globalCounts,
183 DeviceVector&& /*scratch*/ = std::vector<KeyType>{})
184 {
185 gsl::span<const KeyType> leaves(leaves_);
186 std::vector<MPI_Request> treeletRequests;
187 exchangeTreelets(peers_, assignment_, leaves, treelets_, treeletRequests);
188
189 leafCounts_.resize(nNodes(leaves_));
190 if constexpr (HaveGpu<Accelerator>{})
191 {
192 reallocateDestructive(leafCountsAcc_, nNodes(leavesAcc_), 1.01);
193 TreeNodeIndex numLeafNodes = treeData_.numLeafNodes;
194
195 computeNodeCountsGpu(rawPtr(leavesAcc_), rawPtr(leafCountsAcc_), numLeafNodes, particleKeys.begin(),
196 particleKeys.end(), std::numeric_limits<unsigned>::max(), false);
197 memcpyD2H(rawPtr(leafCountsAcc_), numLeafNodes, leafCounts_.data());
198 }
199 else
200 {
201 // local node counts
202 assert(std::is_sorted(particleKeys.begin(), particleKeys.end()));
203 computeNodeCounts(leaves_.data(), leafCounts_.data(), nNodes(leaves_), particleKeys.data(),
204 particleKeys.data() + particleKeys.size(), std::numeric_limits<unsigned>::max(), true);
205 }
206
207 // 1st upsweep with local data
208 counts_.resize(treeData_.numNodes);
209 scatter<TreeNodeIndex>(leafToInternal(treeData_), leafCounts_.data(), counts_.data());
210 upsweep(treeData_.levelRange, treeData_.childOffsets, counts_.data(), NodeCount<unsigned>{});
211
212 // global counts
213 auto globalCountIndices = invertRanges(0, assignment_, nNodes(leaves_));
214
215 // particle counts for leaf nodes in treeLeaves() / leafCounts():
216 // Node indices [firstFocusNode:lastFocusNode] got assigned counts from local particles.
217 // Node index ranges listed in requestIndices got assigned counts from peer ranks.
218 // All remaining indices need to get their counts from the global tree.
219 // They are stored in globalCountIndices.
220 for (auto ip : globalCountIndices)
221 {
222 countRequestParticles<KeyType>(globalTreeLeaves, globalCounts, leaves.subspan(ip.start(), ip.count() + 1),
223 treeData_.prefixes, treeData_.levelRange, gsl::span<unsigned>(counts_));
224 }
225
226 // counts from neighboring peers
227 MPI_Waitall(int(peers_.size()), treeletRequests.data(), MPI_STATUS_IGNORE);
228 constexpr int countTag = static_cast<int>(P2pTags::focusPeerCounts) + 1;
229 exchangeTreeletGeneral(peers_, treelets_, assignment_, gsl::span<const KeyType>(treeData_.prefixes),
230 treeData_.levelRange, leafToInternal(treeData_), gsl::span<unsigned>(counts_), countTag);
231
232 // 2nd upsweep with peer and global data present
233 upsweep(treeData_.levelRange, treeData_.childOffsets, counts_.data(), NodeCount<unsigned>{});
234 gather(leafToInternal(treeData_), counts_.data(), leafCounts_.data());
235
236 if constexpr (HaveGpu<Accelerator>{})
237 {
238 memcpyH2D(leafCounts_.data(), assignment_[myRank_].start(), rawPtr(leafCountsAcc_));
239 memcpyH2D(leafCounts_.data() + assignment_[myRank_].end(), leafCounts_.size() - assignment_[myRank_].end(),
240 rawPtr(leafCountsAcc_) + assignment_[myRank_].end());
241 reallocateDestructive(countsAcc_, counts_.size(), 1.01);
242 memcpyH2D(counts_.data(), counts_.size(), rawPtr(countsAcc_));
243 }
244
245 rebalanceStatus_ |= countsCriterion;
246 }

262 template<class T>
263 void populateGlobal(gsl::span<const KeyType> globalLeaves,
264 gsl::span<const T> localQuantities,
265 gsl::span<T> globalQuantities) const
266 {
267 assert(localQuantities.size() == treeData_.numNodes);
268
269 TreeNodeIndex firstGlobalIdx = findNodeAbove(globalLeaves.data(), globalLeaves.size(), prevFocusStart);
270 TreeNodeIndex lastGlobalIdx = findNodeAbove(globalLeaves.data(), globalLeaves.size(), prevFocusEnd);
271 // make sure that the focus is resolved exactly in the global tree
272 assert(globalLeaves[firstGlobalIdx] == prevFocusStart);
273 assert(globalLeaves[lastGlobalIdx] == prevFocusEnd);
274
275 const KeyType* nodeKeys = treeData_.prefixes.data();
276 const TreeNodeIndex* levelRange = treeData_.levelRange.data();
277
278#pragma omp parallel for schedule(static)
279 for (TreeNodeIndex globalIdx = firstGlobalIdx; globalIdx < lastGlobalIdx; ++globalIdx)
280 {
281 TreeNodeIndex localIdx =
282 locateNode(globalLeaves[globalIdx], globalLeaves[globalIdx + 1], nodeKeys, levelRange);
283 if (localIdx == treeData_.numNodes)
284 {
285 // If the global tree is fully converged, but the locally focused tree is just being built up
286 // for the first time, it's possible that the global tree has a higher resolution than
287 // the focused tree.
288 continue;
289 }
290 assert(decodePlaceholderBit(nodeKeys[localIdx]) == globalLeaves[globalIdx]);
291 assert(decodePrefixLength(nodeKeys[localIdx]) ==
292 3 * treeLevel(globalLeaves[globalIdx + 1] - globalLeaves[globalIdx]));
293 globalQuantities[globalIdx] = localQuantities[localIdx];
294 }
295 }

304 template<class T>
305 void extractGlobal(const KeyType* globalNodeKeys,
306 const TreeNodeIndex* globalLevelRange,
307 gsl::span<const T> globalQuantities,
308 gsl::span<T> localQuantities) const
309 {
310 const KeyType* localLeaves = leaves_.data();
311 const TreeNodeIndex* toInternal = leafToInternal(treeData_).data();
313 auto requestIndices = invertRanges(0, assignment_, treeData_.numLeafNodes);
314 for (auto range : requestIndices)
315 {
317#pragma omp parallel for schedule(static)
318 for (TreeNodeIndex i = range.start(); i < range.end(); ++i)
319 {
320 TreeNodeIndex globalIndex =
321 locateNode(localLeaves[i], localLeaves[i + 1], globalNodeKeys, globalLevelRange);
322 TreeNodeIndex internalIdx = toInternal[i];
323 localQuantities[internalIdx] = globalQuantities[globalIndex];
324 }
325 }

328 template<class Tm, class DeviceVector = std::vector<LocalIndex>>
329 void updateCenters(const RealType* x,
330 const RealType* y,
331 const RealType* z,
332 const Tm* m,
333 const SpaceCurveAssignment& assignment,
334 const Octree<KeyType>& globalTree,
335 const Box<RealType>& box,
336 DeviceVector&& scratch1 = std::vector<LocalIndex>{},
337 DeviceVector&& scratch2 = std::vector<LocalIndex>{})
338 {
339 TreeNodeIndex firstIdx = assignment_[myRank_].start();
340 TreeNodeIndex lastIdx = assignment_[myRank_].end();
341 OctreeView<const KeyType> octree = octreeViewAcc();
342 TreeNodeIndex numNodes = octree.numInternalNodes + octree.numLeafNodes;
343
344 globalCenters_.resize(globalTree.numTreeNodes());
345 centers_.resize(numNodes);
346 reallocateDestructive(centersAcc_, centers_.size(), 1.01);
347
348 if constexpr (HaveGpu<Accelerator>{})
349 {
350 static_assert(IsDeviceVector<std::decay_t<DeviceVector>>{});
351 size_t bytesLayout = (octree.numLeafNodes + 1) * sizeof(LocalIndex);
352 size_t osz1 = reallocateBytes(scratch1, bytesLayout);
353 auto* d_layout = reinterpret_cast<LocalIndex*>(rawPtr(scratch1));
354
355 fillGpu(d_layout, d_layout + octree.numLeafNodes + 1, LocalIndex(0));
356 exclusiveScanGpu(rawPtr(leafCountsAcc_) + firstIdx, rawPtr(leafCountsAcc_) + lastIdx + 1,
357 d_layout + firstIdx);
358 computeLeafSourceCenterGpu(x, y, z, m, octree.leafToInternal + octree.numInternalNodes, octree.numLeafNodes,
359 d_layout, rawPtr(centersAcc_));
361 upsweepCentersGpu(maxTreeLevel<KeyType>{}, treeData_.levelRange.data(), octree.childOffsets,
362 rawPtr(centersAcc_));
363 memcpyD2H(rawPtr(centersAcc_), numNodes, centers_.data());
364
365 reallocateDevice(scratch1, osz1, 1.0);
366 }
367 else
368 {
370 std::vector<LocalIndex> layout(leafCounts_.size() + 1, 0);
371 std::exclusive_scan(leafCounts_.begin() + firstIdx, leafCounts_.begin() + lastIdx + 1,
372 layout.begin() + firstIdx, 0);
373#pragma omp parallel for schedule(static)
374 for (TreeNodeIndex leafIdx = 0; leafIdx < treeData_.numLeafNodes; ++leafIdx)
375 {
377 TreeNodeIndex nodeIdx = octree.leafToInternal[octree.numInternalNodes + leafIdx];
378 centers_[nodeIdx] = massCenter<RealType>(x, y, z, m, layout[leafIdx], layout[leafIdx + 1]);
379 }
381 upsweep(treeData_.levelRange, treeData_.childOffsets, centers_.data(), CombineSourceCenter<RealType>{});
382 }
383
385 peerExchange<SourceCenterType<RealType>>(centers_, static_cast<int>(P2pTags::focusPeerCenters));
387 std::vector<SourceCenterType<RealType>> globalLeafCenters(globalTree.numLeafNodes());
388 populateGlobal<SourceCenterType<RealType>>(globalTree.treeLeaves(), centers_, globalLeafCenters);
389 mpiAllreduce(MPI_IN_PLACE, globalLeafCenters.data(), globalLeafCenters.size(), MPI_SUM);
390 scatter(globalTree.internalOrder(), globalLeafCenters.data(), globalCenters_.data());
391 upsweep(globalTree.levelRange(), globalTree.childOffsets(), globalCenters_.data(),
392 CombineSourceCenter<RealType>{});
393 extractGlobal<SourceCenterType<RealType>>(globalTree.nodeKeys().data(), globalTree.levelRange().data(),
394 globalCenters_, centers_);
395
397 upsweep(treeData_.levelRange, treeData_.childOffsets, centers_.data(), CombineSourceCenter<RealType>{});
399 setMac<RealType, KeyType>(treeData_.prefixes, centers_, 1.0 / theta_, box);
400
401 if constexpr (HaveGpu<Accelerator>{}) { memcpyH2D(centers_.data(), centers_.size(), rawPtr(centersAcc_)); }
402 // else { omp_copy(centers_.begin(), centers_.end(), centersAcc_.begin()); }
403 }

412 void updateMinMac(const Box<RealType>& box,
413 const SpaceCurveAssignment& assignment,
414 gsl::span<const KeyType> globalTreeLeaves,
415 float invThetaEff)
416 {
417 centers_.resize(treeData_.numNodes);

469 template<class DeviceVector = std::vector<KeyType>>
470 void converge(const Box<RealType>& box,
471 gsl::span<const KeyType> particleKeys,
472 gsl::span<const int> peers,
473 const SpaceCurveAssignment& assignment,
474 gsl::span<const KeyType> globalTreeLeaves,
475 gsl::span<const unsigned> globalCounts,
476 float invThetaEff,
477 DeviceVector&& scratch = std::vector<KeyType>{})
478 {
479 int converged = 0;
480 while (converged != numRanks_)
481 {
482 converged = updateTree(peers, assignment, globalTreeLeaves);
483 updateCounts(particleKeys, globalTreeLeaves, globalCounts, scratch);
484 updateMinMac(box, assignment, globalTreeLeaves, invThetaEff);
485 MPI_Allreduce(MPI_IN_PLACE, &converged, 1, MPI_INT, MPI_SUM, MPI_COMM_WORLD);
486 }
487 }

671template<class Q, class KeyType, class T, class F, class Accelerator, class... UArgs>
672void globalFocusExchange(const Octree<KeyType>& globalOctree,
673 const FocusedOctree<KeyType, T, Accelerator>& focusTree,
674 gsl::span<Q> quantities,
675 F&& upsweepFunction,
676 UArgs&&... upsweepArgs)
677{
678 TreeNodeIndex numGlobalLeaves = globalOctree.numLeafNodes();
679 std::vector<Q> globalLeafQuantities(numGlobalLeaves);
680 focusTree.template populateGlobal<Q>(globalOctree.treeLeaves(), quantities, globalLeafQuantities);
681
683 mpiAllreduce(MPI_IN_PLACE, globalLeafQuantities.data(), numGlobalLeaves, MPI_SUM);
684
685 std::vector<Q> globalQuantities(globalOctree.numTreeNodes());
686 scatter(globalOctree.internalOrder(), globalLeafQuantities.data(), globalQuantities.data());
688 upsweepFunction(globalOctree.levelRange(), globalOctree.childOffsets(), globalQuantities.data(), upsweepArgs...);
689
691 focusTree.template extractGlobal<Q>(globalOctree.nodeKeys().data(), globalOctree.levelRange().data(),
692 globalQuantities, quantities);
693}
