62template<class KeyType>
63struct CombinedUpdate
64{
80 static bool updateFocus(OctreeData<KeyType, CpuTag>& tree,
81 std::vector<KeyType>& leaves,
82 unsigned bucketSize,
83 KeyType focusStart,
84 KeyType focusEnd,
85 gsl::span<const KeyType> mandatoryKeys,
86 gsl::span<const unsigned> counts,
87 gsl::span<const char> macs)
88 {
89 [[maybe_unused]] TreeNodeIndex numNodes = tree.numLeafNodes + tree.numInternalNodes;
90 assert(TreeNodeIndex(counts.size()) == numNodes);
91 assert(TreeNodeIndex(macs.size()) == numNodes);
92 assert(TreeNodeIndex(tree.internalToLeaf.size()) >= numNodes);
93
94 // take op decision per node
95 gsl::span<TreeNodeIndex> nodeOpsAll(tree.internalToLeaf);
96 rebalanceDecisionEssential<KeyType>(tree.prefixes, tree.childOffsets.data(), tree.parents.data(), counts.data(),
97 macs.data(), focusStart, focusEnd, bucketSize, nodeOpsAll.data());
98
99 std::vector<KeyType> allMandatoryKeys{focusStart, focusEnd};
100 std::copy(mandatoryKeys.begin(), mandatoryKeys.end(), std::back_inserter(allMandatoryKeys));
101 auto status = enforceKeys<KeyType>(allMandatoryKeys, tree.prefixes.data(), tree.childOffsets.data(),
102 tree.parents.data(), nodeOpsAll.data());
103
104 bool converged = protectAncestors<KeyType>(tree.prefixes, tree.parents.data(), nodeOpsAll.data());
105
106 // extract leaf decision, using childOffsets at temp storage, require +1 for exclusive scan last element
107 assert(tree.childOffsets.size() >= size_t(tree.numLeafNodes + 1));
108 gsl::span<TreeNodeIndex> nodeOps(tree.childOffsets.data(), tree.numLeafNodes + 1);
109 gather(leafToInternal(tree), nodeOpsAll.data(), nodeOps.data());
110
111 if (status == ResolutionStatus::cancelMerge)
112 {
113 converged = std::all_of(nodeOps.begin(), nodeOps.end() - 1, [](TreeNodeIndex i) { return i == 1; });
114 }
115 else if (status == ResolutionStatus::rebalance) { converged = false; }
116
117 // carry out rebalance based on nodeOps
118 auto& newLeaves = tree.prefixes;
119 rebalanceTree(leaves, newLeaves, nodeOps.data());
120
121 // if rebalancing couldn't introduce the mandatory keys, we force-inject them now into the tree
122 if (status == ResolutionStatus::failed)
123 {
124 converged = false;
125 injectKeys<KeyType>(newLeaves, allMandatoryKeys);
126 }
127
128 swap(newLeaves, leaves);
129 tree.resize(nNodes(leaves));
130 updateInternalTree<KeyType>(leaves, tree.data());
131
132 return converged;
133 }

151 template<class Alloc>
152 static bool updateFocusGpu(OctreeData<KeyType, GpuTag>& tree,
153 thrust::device_vector<KeyType, Alloc>& leaves,
154 unsigned bucketSize,
155 KeyType focusStart,
156 KeyType focusEnd,
157 gsl::span<const KeyType> mandatoryKeys,
158 gsl::span<const unsigned> counts,
159 gsl::span<const char> macs)
160 {
161 [[maybe_unused]] TreeNodeIndex numNodes = tree.numLeafNodes + tree.numInternalNodes;
162 assert(TreeNodeIndex(counts.size()) == numNodes);
163 assert(TreeNodeIndex(macs.size()) == numNodes);
164 assert(TreeNodeIndex(tree.internalToLeaf.size()) >= numNodes);
165
166 // take op decision per node
167 gsl::span<TreeNodeIndex> nodeOpsAll(rawPtr(tree.internalToLeaf), numNodes);
168 rebalanceDecisionEssentialGpu(rawPtr(tree.prefixes), rawPtr(tree.childOffsets), rawPtr(tree.parents),
169 counts.data(), macs.data(), focusStart, focusEnd, bucketSize, nodeOpsAll.data(),
170 numNodes);
171
172 auto status = ResolutionStatus::converged;
173 if (!mandatoryKeys.empty())
174 {
175 thrust::device_vector<KeyType, Alloc> d_mandatoryKeys;
176 reallocate(d_mandatoryKeys, mandatoryKeys.size(), 1.0);
177 memcpyH2D(mandatoryKeys.data(), mandatoryKeys.size(), rawPtr(d_mandatoryKeys));
178 status = enforceKeysGpu(rawPtr(d_mandatoryKeys), d_mandatoryKeys.size(), rawPtr(tree.prefixes),
179 rawPtr(tree.childOffsets), rawPtr(tree.parents), nodeOpsAll.data());
180 }
181 bool converged = protectAncestorsGpu(rawPtr(tree.prefixes), rawPtr(tree.parents), nodeOpsAll.data(), numNodes);
182
183 // extract leaf decision, using childOffsets as temp storage
184 assert(tree.childOffsets.size() >= size_t(tree.numLeafNodes + 1));
185 gsl::span<TreeNodeIndex> nodeOps(rawPtr(tree.childOffsets), tree.numLeafNodes + 1);
186 gatherGpu(leafToInternal(tree).data(), nNodes(leaves), nodeOpsAll.data(), nodeOps.data());
187
188 if (status == ResolutionStatus::cancelMerge)
189 {
190 converged = countGpu(nodeOps.begin(), nodeOps.end() - 1, 1) == tree.numLeafNodes;
191 }
192 else if (status == ResolutionStatus::rebalance) { converged = false; }
193
194 exclusiveScanGpu(nodeOps.data(), nodeOps.data() + nodeOps.size(), nodeOps.data());
195 TreeNodeIndex newNumLeafNodes;
196 memcpyD2H(nodeOps.data() + nodeOps.size() - 1, 1, &newNumLeafNodes);
197
198 // carry out rebalance based on nodeOps
199 auto& newLeaves = tree.prefixes;
200 reallocateDestructive(newLeaves, newNumLeafNodes + 1, 1.01);
201 rebalanceTreeGpu(rawPtr(leaves), nNodes(leaves), newNumLeafNodes, nodeOps.data(), rawPtr(newLeaves));
202
203 // if rebalancing couldn't introduce the mandatory keys, we force-inject them now into the tree
204 if (status == ResolutionStatus::failed)
205 {
206 converged = false;
207
208 std::vector<KeyType> hostLeaves(newLeaves.size());
209 memcpyD2H(rawPtr(newLeaves), newLeaves.size(), hostLeaves.data());
210
211 injectKeys<KeyType>(hostLeaves, mandatoryKeys);
212 reallocateDestructive(newLeaves, hostLeaves.size(), 1.01);
213 memcpyH2D(hostLeaves.data(), newLeaves.size(), rawPtr(newLeaves));
214 }
215
216 swap(newLeaves, leaves);
217 tree.resize(nNodes(leaves));
218 buildOctreeGpu(rawPtr(leaves), tree.data());
219
220 return converged;
221 }

228template<class KeyType>
229class FocusedOctreeSingleNode
230{
231 using CB = CombinedUpdate<KeyType>;
232
233public:
234 FocusedOctreeSingleNode(unsigned bucketSize, float theta)
235 : theta_(theta)
236 , bucketSize_(bucketSize)
237 , counts_{bucketSize + 1}
238 , macs_{1}
239 {
240 leaves_ = std::vector<KeyType>{0, nodeRange<KeyType>(0)};
241 tree_.resize(nNodes(leaves_));
242 updateInternalTree<KeyType>(leaves_, tree_.data());
243 }
244
246 template<class T>
247 bool update(const Box<T>& box,
248 gsl::span<const KeyType> particleKeys,
249 KeyType focusStart,
250 KeyType focusEnd,
251 gsl::span<const KeyType> mandatoryKeys)
252 {
253 bool converged =
254 CB::updateFocus(tree_, leaves_, bucketSize_, focusStart, focusEnd, mandatoryKeys, counts_, macs_);
255
256 std::vector<Vec4<T>> centers_(tree_.numNodes);
257 float invThetaEff = 1.0f / theta_ + 0.5;
258
259#pragma omp parallel for schedule(static)
260 for (TreeNodeIndex i = 0; i < tree_.numNodes; ++i)
261 {
263 centers_[i] = computeMinMacR2(tree_.prefixes[i], invThetaEff, box);
264 }
265
266 macs_.resize(tree_.numNodes);
267 markMacs(tree_.data(), centers_.data(), box, focusStart, focusEnd, macs_.data());
268
269 leafCounts_.resize(nNodes(leaves_));
270 computeNodeCounts(leaves_.data(), leafCounts_.data(), nNodes(leaves_), particleKeys.data(),
271 particleKeys.data() + particleKeys.size(), std::numeric_limits<unsigned>::max(), true);
272
273 counts_.resize(tree_.numNodes);
274 scatter(leafToInternal(tree_), leafCounts_.data(), counts_.data());
275 upsweep(tree_.levelRange, tree_.childOffsets, counts_.data(), NodeCount<unsigned>{});
276
277 return converged;
278 }
279
280 gsl::span<const KeyType> treeLeaves() const { return leaves_; }
281 gsl::span<const unsigned> leafCounts() const { return leafCounts_; }
282
283private:
285 float theta_;
286 unsigned bucketSize_;
287
288 OctreeData<KeyType, CpuTag> tree_;
289 std::vector<KeyType> leaves_;
290
292 std::vector<unsigned> leafCounts_;
293 std::vector<unsigned> counts_;
295 std::vector<char> macs_;
296};
