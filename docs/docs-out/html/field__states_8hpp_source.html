161 auto checkTypesMatch = [](const auto* var1, const auto* var2)
162 {
163 using Type1 = std::decay_t<decltype(*var1)>;
164 using Type2 = std::decay_t<decltype(*var2)>;
165 return std::is_same_v<Type1, Type2>;
166 };
167
168 auto swapFields = [](auto* varPtr1, auto* varPtr2)
169 {
170 using Type1 = std::decay_t<decltype(*varPtr1)>;
171 using Type2 = std::decay_t<decltype(*varPtr2)>;
172 if constexpr (std::is_same_v<Type1, Type2>) { swap(*varPtr1, *varPtr2); }
173 };
174
175 for (size_t i = 0; i < fieldStates_.size(); ++i)
176 {
177 if (fieldStates_[i] == State::released)
178 {
179 bool typesMatch = std::visit(checkTypesMatch, data_[i], data_[fieldIdx]);
180 if (typesMatch)
181 {
182 std::visit(swapFields, data_[i], data_[fieldIdx]);
183 fieldStates_[i] = State::unused;
184 fieldStates_[fieldIdx] = State::dependent;
185 return;
186 }
187 }
188 }
189 throw std::runtime_error("Could not acquire field " + std::string(DataType::fieldNames[fieldIdx]) +
190 ". No suitable field available");
191 }
