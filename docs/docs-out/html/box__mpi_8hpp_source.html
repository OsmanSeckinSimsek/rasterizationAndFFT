48template<class T>
49struct MinMax
50{
51 std::tuple<T, T> operator()(const T* start, const T* end)
52 {
53 assert(end >= start);
54
55 T minimum = INFINITY;
56 T maximum = -INFINITY;
57
58#pragma omp parallel for reduction(min : minimum) reduction(max : maximum)
59 for (size_t pi = 0; pi < std::size_t(end - start); pi++)
60 {
61 T value = start[pi];
62 minimum = std::min(minimum, value);
63 maximum = std::max(maximum, value);
64 }
65
66 return std::make_tuple(minimum, maximum);
67 }
68};

83template<class T, class Op = MinMax<T>>
84auto makeGlobalBox(const T* x, const T* y, const T* z, size_t numElements, const Box<T>& previousBox = Box<T>(0, 1))
85{
86 bool pbcX = (previousBox.boundaryX() == BoundaryType::periodic);
87 bool pbcY = (previousBox.boundaryY() == BoundaryType::periodic);
88 bool pbcZ = (previousBox.boundaryZ() == BoundaryType::periodic);
89
90 std::array<T, 6> extrema;
91 std::tie(extrema[0], extrema[1]) =
92 pbcX ? std::make_tuple(previousBox.xmin(), previousBox.xmax()) : Op{}(x, x + numElements);
93 std::tie(extrema[2], extrema[3]) =
94 pbcY ? std::make_tuple(previousBox.ymin(), previousBox.ymax()) : Op{}(y, y + numElements);
95 std::tie(extrema[4], extrema[5]) =
96 pbcZ ? std::make_tuple(previousBox.zmin(), previousBox.zmax()) : Op{}(z, z + numElements);
97
98 if (!pbcX || !pbcY || !pbcZ)
99 {
100 extrema[1] = -extrema[1];
101 extrema[3] = -extrema[3];
102 extrema[5] = -extrema[5];
103 MPI_Allreduce(MPI_IN_PLACE, extrema.data(), extrema.size(), MpiType<T>{}, MPI_MIN, MPI_COMM_WORLD);
104 extrema[1] = -extrema[1];
105 extrema[3] = -extrema[3];
106 extrema[5] = -extrema[5];
107 }
108
109 return Box<T>{extrema[0],
110 extrema[1],
111 extrema[2],
112 extrema[3],
113 extrema[4],
114 extrema[5],
115 previousBox.boundaryX(),
116 previousBox.boundaryY(),
117 previousBox.boundaryZ()};
118}
