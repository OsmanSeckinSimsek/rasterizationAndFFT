41template<class T>
42using SourceCenterType = util::array<T, 4>;

45template<class T>
46HOST_DEVICE_FUN void addBody(SourceCenterType<T>& center, const SourceCenterType<T>& source)
47{
48 T weight = source[3];
49
50 center[0] += weight * source[0];
51 center[1] += weight * source[1];
52 center[2] += weight * source[2];
53 center[3] += weight;
54}
55
57template<class T>
58HOST_DEVICE_FUN SourceCenterType<T> normalizeMass(SourceCenterType<T> center)
59{
60 T invM = (center[3] != T(0.0)) ? T(1.0) / center[3] : T(0.0);
61 center[0] *= invM;
62 center[1] *= invM;
63 center[2] *= invM;
64
65 return center;
66}

69template<class Ts, class Tc, class Tm>
70HOST_DEVICE_FUN SourceCenterType<Ts>
71massCenter(const Tc* x, const Tc* y, const Tc* z, const Tm* m, LocalIndex first, LocalIndex last)
72{
73 SourceCenterType<Ts> center{0, 0, 0, 0};
74 for (LocalIndex i = first; i < last; ++i)
75 {
76 addBody(center, SourceCenterType<Ts>{Ts(x[i]), Ts(y[i]), Ts(z[i]), Ts(m[i])});
77 }
78
79 return normalizeMass(center);
80}

83template<class T>
84struct CombineSourceCenter
85{
86 HOST_DEVICE_FUN
87 SourceCenterType<T> operator()(TreeNodeIndex /*nodeIdx*/, TreeNodeIndex child, const SourceCenterType<T>* centers)
88 {
89 SourceCenterType<T> center{0, 0, 0, 0};
90
91 for (TreeNodeIndex i = child; i < child + 8; ++i)
92 {
93 addBody(center, centers[i]);

129template<class T, class KeyType>
130void setMac(gsl::span<const KeyType> nodeKeys,
131 gsl::span<SourceCenterType<T>> centers,
132 float invTheta,
133 const Box<T>& box)
134{
135#pragma omp parallel for schedule(static)
136 for (size_t i = 0; i < nodeKeys.size(); ++i)
137 {
138 Vec4<T> center = centers[i];
139 T mac = computeVecMacR2(nodeKeys[i], util::makeVec3(center), invTheta, box);
140 centers[i][3] = (center[3] != T(0)) ? mac : T(0);
141 }
142}

145template<class KeyType, class T>
146void nodeFpCenters(gsl::span<const KeyType> prefixes, Vec3<T>* centers, Vec3<T>* sizes, const Box<T>& box)
147{
148#pragma omp parallel for schedule(static)
149 for (size_t i = 0; i < prefixes.size(); ++i)
150 {
151 KeyType prefix = prefixes[i];
152 KeyType startKey = decodePlaceholderBit(prefix);
153 unsigned level = decodePrefixLength(prefix) / 3;
154 auto nodeBox = sfcIBox(sfcKey(startKey), level);
155 util::tie(centers[i], sizes[i]) = centerAndSize<KeyType>(nodeBox, box);
156 }
157}
