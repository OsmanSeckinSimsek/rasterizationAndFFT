51HOST_DEVICE_FUN
52constexpr unsigned expandBits(unsigned v)
53{
54 v &= 0x000003ffu; // discard bit higher 10
55 v = (v * 0x00010001u) & 0xFF0000FFu;
56 v = (v * 0x00000101u) & 0x0F00F00Fu;
57 v = (v * 0x00000011u) & 0xC30C30C3u;
58 v = (v * 0x00000005u) & 0x49249249u;
59 return v;
60}

65HOST_DEVICE_FUN
66constexpr unsigned compactBits(unsigned v)
67{
68 v &= 0x09249249u;
69 v = (v ^ (v >> 2u)) & 0x030c30c3u;
70 v = (v ^ (v >> 4u)) & 0x0300f00fu;
71 v = (v ^ (v >> 8u)) & 0xff0000ffu;
72 v = (v ^ (v >> 16u)) & 0x000003ffu;
73 return v;
74}

77HOST_DEVICE_FUN
78constexpr uint64_t expandBits(uint64_t v)
79{
80 uint64_t x = v & 0x1fffffu; // discard bits higher 21
81 x = (x | x << 32u) & 0x001f00000000fffflu;
82 x = (x | x << 16u) & 0x001f0000ff0000fflu;
83 x = (x | x << 8u) & 0x100f00f00f00f00flu;
84 x = (x | x << 4u) & 0x10c30c30c30c30c3lu;
85 x = (x | x << 2u) & 0x1249249249249249lu;
86 return x;
87}

92HOST_DEVICE_FUN
93constexpr uint64_t compactBits(uint64_t v)
94{
95 v &= 0x1249249249249249lu;
96 v = (v ^ (v >> 2u)) & 0x10c30c30c30c30c3lu;
97 v = (v ^ (v >> 4u)) & 0x100f00f00f00f00flu;
98 v = (v ^ (v >> 8u)) & 0x001f0000ff0000fflu;
99 v = (v ^ (v >> 16u)) & 0x001f00000000fffflu;
100 v = (v ^ (v >> 32u)) & 0x00000000001ffffflu;
101 return v;
102}

111template<class KeyType>
112constexpr HOST_DEVICE_FUN std::enable_if_t<std::is_unsigned<KeyType>{}, KeyType>
113iMorton(unsigned ix, unsigned iy, unsigned iz) noexcept
114{
115 assert(ix < (1u << maxTreeLevel<KeyType>{}));
116 assert(iy < (1u << maxTreeLevel<KeyType>{}));
117 assert(iz < (1u << maxTreeLevel<KeyType>{}));
118
119 KeyType xx = detail::expandBits(KeyType(ix));
120 KeyType yy = detail::expandBits(KeyType(iy));
121 KeyType zz = detail::expandBits(KeyType(iz));
122
123 // interleave the x, y, z components
124 return xx * 4 + yy * 2 + zz;
125}

134template<class KeyType>
135constexpr HOST_DEVICE_FUN KeyType iMorton(unsigned ix, unsigned iy, unsigned iz, unsigned treeLevel)
136{
137 assert(treeLevel <= maxTreeLevel<KeyType>{});
138 unsigned shifts = maxTreeLevel<KeyType>{} - treeLevel;
139 return iMorton<KeyType>(ix << shifts, iy << shifts, iz << shifts);
140}

177template<class KeyType>
178HOST_DEVICE_FUN IBox mortonIBox(KeyType keyStart, unsigned level) noexcept
179{
180 assert(level <= maxTreeLevel<KeyType>{});
181 unsigned cubeLength = (1u << (maxTreeLevel<KeyType>{} - level));
182 auto [ix, iy, iz] = decodeMorton(keyStart);
183 return IBox(ix, ix + cubeLength, iy, iy + cubeLength, iz, iz + cubeLength);
184}
