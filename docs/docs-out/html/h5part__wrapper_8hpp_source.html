49std::string H5PartTypeToString(h5part_int64_t type)
50{
51 if (type == H5PART_FLOAT64) { return "C++: double / python: np.float64"; }
52 if (type == H5PART_FLOAT32) { return "C++: float / python: np.float32"; }
53 if (type == H5PART_INT32) { return "C++: int / python: np.int32"; }
54 if (type == H5PART_INT64) { return "C++: int64_t / python: np.int64"; }
55 if (type == H5PART_CHAR) { return "C++: char / python: np.int8"; }
56
57 return "H5PART_UNKNOWN";
58}

108std::vector<std::string> datasetNames(H5PartFile* h5_file)
109{
110 auto numSets = H5PartGetNumDatasets(h5_file);
111
112 std::vector<std::string> setNames(numSets);
113 for (int64_t fi = 0; fi < numSets; ++fi)
114 {
115 int maxlen = 256;
116 char fieldName[maxlen];
117 H5PartGetDatasetName(h5_file, fi, fieldName, maxlen);
118 setNames[fi] = std::string(fieldName);
119 }
120
121 return setNames;
122}

175template<class ExtractType, class Reader, class Info>
176void readAttribute(ExtractType* attr, int attrSizeBuf, int attrIndex, Reader&& readAttrib, Info&& readAttribInfo)
177{
178 using IoTuple = util::Reduce<std::tuple, H5PartTypes>;
179
180 h5part_int64_t typeId, attrSizeFile;
181 char attrName[256];
182 readAttribInfo(attrIndex, attrName, 256, &typeId, &attrSizeFile);
183 bool breakLoop = false;
184
185 if (attrSizeBuf != attrSizeFile)
186 {
187 throw std::runtime_error("Attribute " + std::string(attrName) + " size mismatch: in file " +
188 std::to_string(attrSizeFile) + ", but provided buffer has size " +
189 std::to_string(attrSizeBuf) + "\n");
190 }
191
192 auto readTypesafe = [&](auto dummyValue)
193 {
194 using TypeInFile = std::decay_t<decltype(dummyValue)>;
195 if (fileutils::H5PartType<TypeInFile>{} == typeId && not breakLoop)
196 {
197 std::vector<TypeInFile> attrBuf(attrSizeFile);
198 if (readAttrib(attrName, attrBuf.data()) != H5PART_SUCCESS)
199 {
200 throw std::runtime_error("Could not read attribute " + std::string(attrName) + "\n");
201 }
202
203 bool bothFloating = std::is_floating_point_v<TypeInFile> && std::is_floating_point_v<ExtractType>;
204 bool extractToCommonType = std::is_same_v<std::common_type_t<TypeInFile, ExtractType>, ExtractType>;
205 if (bothFloating || extractToCommonType) { std::copy(attrBuf.begin(), attrBuf.end(), attr); }
206 else
207 {
208 int64_t memTypeId = fileutils::H5PartType<ExtractType>{};
209 throw std::runtime_error("Reading attribute " + std::string(attrName) +
210 " failed: " + "type in file is " + fileutils::H5PartTypeToString(typeId) +
211 ", but supplied buffer type is " + fileutils::H5PartTypeToString(memTypeId) +
212 "\n");
213 }
214 breakLoop = true;
215 }
216 };
217 util::for_each_tuple(readTypesafe, IoTuple{});
218}

323H5PartFile* openH5Part(const std::string& path, h5part_int64_t mode, MPI_Comm comm)
324{
325 const char* h5_fname = path.c_str();
326 H5PartFile* h5_file = nullptr;
327
328#ifdef H5PART_PARALLEL_IO
329 h5_file = H5PartOpenFileParallel(h5_fname, mode, comm);
330#else
331 int numRanks;
332 MPI_Comm_size(comm, &numRanks);
333 if (numRanks > 1)
334 {
335 throw std::runtime_error("Cannot open HDF5 file on multiple ranks without parallel HDF5 support\n");
336 }
337 h5_file = H5PartOpenFile(h5_fname, mode);
338#endif
339
340 return h5_file;
341}
