58template<class KeyType>
59constexpr HOST_DEVICE_FUN inline std::enable_if_t<std::is_unsigned_v<KeyType>, KeyType>
60iHilbert(unsigned px, unsigned py, unsigned pz) noexcept
61{
62 assert(px < (1u << maxTreeLevel<KeyType>{}));
63 assert(py < (1u << maxTreeLevel<KeyType>{}));
64 assert(pz < (1u << maxTreeLevel<KeyType>{}));
65
66#if !defined(__CUDA_ARCH__) && !defined(__HIP_DEVICE_COMPILE__)
67 constexpr unsigned mortonToHilbert[8] = {0, 1, 3, 2, 7, 6, 4, 5};
68#endif
69
70 KeyType key = 0;
71
72 for (int level = maxTreeLevel<KeyType>{} - 1; level >= 0; --level)
73 {
74 unsigned xi = (px >> level) & 1u;
75 unsigned yi = (py >> level) & 1u;
76 unsigned zi = (pz >> level) & 1u;
77
78 // append 3 bits to the key
79 unsigned octant = (xi << 2) | (yi << 1) | zi;
80#if defined(__CUDA_ARCH__) || defined(__HIP_DEVICE_COMPILE__)
81 key = (key << 3) + mortonToHilbertDevice[octant];
82#else
83 key = (key << 3) + mortonToHilbert[octant];
84#endif
85
86 // turn px, py and pz
87 px ^= -(xi & ((!yi) | zi));
88 py ^= -((xi & (yi | zi)) | (yi & (!zi)));
89 pz ^= -((xi & (!yi) & (!zi)) | (yi & (!zi)));
90
91 if (zi)
92 {
93 // cyclic rotation
94 unsigned pt = px;
95 px = py;
96 py = pz;
97 pz = pt;
98 }
99 else if (!yi)
100 {
101 // swap x and z
102 unsigned pt = px;
103 px = pz;
104 pz = pt;
105 }
106 }
107
108 return key;
109}

118template<class KeyType>
119constexpr HOST_DEVICE_FUN inline std::enable_if_t<std::is_unsigned_v<KeyType>, KeyType> iHilbert2D(unsigned px,
120 unsigned py) noexcept
121{
122 assert(px < (1u << maxTreeLevel<KeyType>{}));
123 assert(py < (1u << maxTreeLevel<KeyType>{}));
124
125 unsigned xi, yi;
126 unsigned temp;
127 KeyType key = 0;
128
129 for (int level = maxTreeLevel<KeyType>{} - 1; level >= 0; level--)
130 {
131 xi = (px >> level) & 1u; // Get bit level of x.
132 yi = (py >> level) & 1u; // Get bit level of y.
133
134 if (yi == 0)
135 {
136 temp = px; // Swap x and y and,
137 px = py ^ (-xi); // if xi = 1,
138 py = temp ^ (-xi); // complement them.
139 }
140 key = 4 * key + 2 * xi + (xi ^ yi); // Append two bits to key.
141 }
142 return key;
143}

146template<class KeyType>
147HOST_DEVICE_FUN inline util::tuple<unsigned, unsigned, unsigned> decodeHilbert(KeyType key) noexcept
148{
149 unsigned px = 0;
150 unsigned py = 0;
151 unsigned pz = 0;
152
153 for (unsigned level = 0; level < maxTreeLevel<KeyType>{}; ++level)
154 {
155 unsigned octant = (key >> (3 * level)) & 7u;
156 const unsigned xi = octant >> 2u;
157 const unsigned yi = (octant >> 1u) & 1u;
158 const unsigned zi = octant & 1u;
159
160 if (yi ^ zi)
161 {
162 // cyclic rotation
163 unsigned pt = px;
164 px = pz;
165 pz = py;
166 py = pt;
167 }
168 else if ((!xi & !yi & !zi) || (xi & yi & zi))
169 {
170 // swap x and z
171 unsigned pt = px;
172 px = pz;
173 pz = pt;
174 }
175
176 // turn px, py and pz
177 unsigned mask = (1 << level) - 1;
178 px ^= mask & (-(xi & (yi | zi)));
179 py ^= mask & (-((xi & ((!yi) | (!zi))) | ((!xi) & yi & zi)));
180 pz ^= mask & (-((xi & (!yi) & (!zi)) | (yi & zi)));
181
182 // append 1 bit to the positions
183 px |= (xi << level);
184 py |= ((xi ^ yi) << level);
185 pz |= ((yi ^ zi) << level);
186 }
187
188 return {px, py, pz};
189}

191// Lam and Shapiro inverse function of hilbert
192template<class KeyType>
193HOST_DEVICE_FUN inline util::tuple<unsigned, unsigned> decodeHilbert2D(KeyType key) noexcept
194{
195 unsigned sa, sb;
196 unsigned x = 0, y = 0, temp = 0;
197 unsigned order = maxTreeLevel<KeyType>{};
198
199 for (unsigned level = 0; level < 2 * order; level += 2)
200 {
201 // Get bit level+1 of key.
202 sa = (key >> (level + 1)) & 1;
203 // Get bit level of key.
204 sb = (key >> level) & 1;
205 if ((sa ^ sb) == 0)
206 {
207 // If sa,sb = 00 or 11,
208 temp = x;
209 // swap x and y,
210 x = y ^ (-sa);
211 // and if sa = 1,
212 y = temp ^ (-sa);
213 // complement them.
214 }
215 x = (x >> 1) | (sa << 31); // Prepend sa to x and
216 y = (y >> 1) | ((sa ^ sb) << 31); // (sa ^ sb) to y.
217 }
218 unsigned px = x >> (32 - order);
219 // Right-adjust x and y
220 unsigned py = y >> (32 - order);
221 // and return them to
222 return {px, py};
223}

226template<class KeyType>
227HOST_DEVICE_FUN inline util::tuple<unsigned, unsigned> decodeHilbert2DConstant(KeyType key) noexcept
228{
229 unsigned order = maxTreeLevel<KeyType>{};
230
231 key = key | (0x55555555 << 2 * order); // Pad key on left with 01
232
233 const unsigned sr = (key >> 1) & 0x55555555; // (no change) groups.
234 unsigned cs = ((key & 0x55555555) + sr) ^ 0x55555555; // Compute complement & swap info in two-bit groups.
235 // Parallel prefix xor op to propagate both complement
236 // and swap info together from left to right (there is
237 // no step "cs ^= cs >> 1", so in effect it computes
238 // two independent parallel prefix operations on two
239 // interleaved sets of sixteen bits).
240 cs = cs ^ (cs >> 2);
241 cs = cs ^ (cs >> 4);
242 cs = cs ^ (cs >> 8);
243 cs = cs ^ (cs >> 16);
244 const unsigned swap = cs & 0x55555555; // Separate the swap and
245 const unsigned comp = (cs >> 1) & 0x55555555; // complement bits.
246
247 unsigned t = (key & swap) ^ comp; // Calculate x and y in
248 key = key ^ sr ^ t ^ (t << 1); // the odd & even bit positions, resp.
249 key = key & ((1 << 2 * order) - 1); // Clear out any junk on the left (unpad).
250
251 // Now "unshuffle" to separate the x and y bits.
252
253 t = (key ^ (key >> 1)) & 0x22222222;
254 key = key ^ t ^ (t << 1);
255 t = (key ^ (key >> 2)) & 0x0C0C0C0C;
256 key = key ^ t ^ (t << 2);
257 t = (key ^ (key >> 4)) & 0x00F000F0;
258 key = key ^ t ^ (t << 4);
259 t = (key ^ (key >> 8)) & 0x0000FF00;
260 key = key ^ t ^ (t << 8);
261
262 unsigned px = key >> 16; // Assign the two halves
263 unsigned py = key & 0xFFFF; // of t to x and y.
264
265 return {px, py};
266}

275template<class KeyType>
276HOST_DEVICE_FUN IBox hilbertIBox(KeyType keyStart, unsigned level) noexcept
277{
278 assert(level <= maxTreeLevel<KeyType>{});
279 constexpr unsigned maxCoord = 1u << maxTreeLevel<KeyType>{};
280 unsigned cubeLength = maxCoord >> level;
281 unsigned mask = ~(cubeLength - 1);
282
283 auto [ix, iy, iz] = decodeHilbert(keyStart);
284
285 // round integer coordinates down to corner closest to origin
286 ix &= mask;
287 iy &= mask;
288 iz &= mask;
289
290 return IBox(ix, ix + cubeLength, iy, iy + cubeLength, iz, iz + cubeLength);
291}
