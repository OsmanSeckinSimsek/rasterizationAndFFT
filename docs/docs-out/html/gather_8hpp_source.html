58template<class InoutIterator, class OutputIterator, class Compare>
59void sort_by_key(InoutIterator keyBegin, InoutIterator keyEnd, OutputIterator valueBegin, Compare compare)
60{
61 using KeyType = std::decay_t<decltype(*keyBegin)>;
62 using ValueType = std::decay_t<decltype(*valueBegin)>;
63 std::size_t n = std::distance(keyBegin, keyEnd);
64
65 // zip the input integer array together with the index sequence
66 std::vector<std::tuple<KeyType, ValueType>, util::DefaultInitAdaptor<std::tuple<KeyType, ValueType>>> keyIndexPairs(
67 n);
68#pragma omp parallel for schedule(static)
69 for (std::size_t i = 0; i < n; ++i)
70 keyIndexPairs[i] = std::make_tuple(keyBegin[i], valueBegin[i]);
71
72 // sort, comparing only the first tuple element
73 std::sort(begin(keyIndexPairs), end(keyIndexPairs),
74 [compare](const auto& t1, const auto& t2) { return compare(std::get<0>(t1), std::get<0>(t2)); });
75
76// extract the resulting ordering and store back the sorted keys
77#pragma omp parallel for schedule(static)
78 for (std::size_t i = 0; i < n; ++i)
79 {
80 keyBegin[i] = std::get<0>(keyIndexPairs[i]);
81 valueBegin[i] = std::get<1>(keyIndexPairs[i]);
82 }
83}
