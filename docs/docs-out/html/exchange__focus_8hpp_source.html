68template<class KeyType>
69void countRequestParticles(gsl::span<const KeyType> leaves,
70 gsl::span<const unsigned> counts,
71 gsl::span<const KeyType> requestLeaves,
72 gsl::span<const KeyType> prefixes,
73 gsl::span<const TreeNodeIndex> levelRange,
74 gsl::span<unsigned> requestCounts)
75{
76#pragma omp parallel for
77 for (size_t i = 0; i < nNodes(requestLeaves); ++i)
78 {
79 KeyType startKey = requestLeaves[i];
80 KeyType endKey = requestLeaves[i + 1];
81
82 size_t startIdx = findNodeBelow(leaves.data(), leaves.size(), startKey);
83 size_t endIdx = findNodeAbove(leaves.data(), leaves.size(), endKey);
84
85 TreeNodeIndex internalIdx = locateNode(startKey, endKey, prefixes.data(), levelRange.data());
86
87 // Nodes in @p leaves must match the request keys exactly, otherwise counts are wrong.
88 // If this assertion fails, it means that the local leaves/counts does not have the required
89 // resolution to answer the incoming count request of [startKey:endKey] precisely, which means that
90 // focusTransfer didn't work correctly
91 assert(startKey == leaves[startIdx]);
92 assert(endKey == leaves[endIdx]);
93
94 uint64_t internalCount = std::accumulate(counts.begin() + startIdx, counts.begin() + endIdx, uint64_t(0));
95 requestCounts[internalIdx] = std::min(uint64_t(std::numeric_limits<unsigned>::max()), internalCount);
96 }
97}

113template<class KeyType>
114void exchangeTreelets(gsl::span<const int> peerRanks,
115 gsl::span<const IndexPair<TreeNodeIndex>> focusAssignment,
116 gsl::span<const KeyType> localLeaves,
117 std::vector<std::vector<KeyType>>& peerTrees,
118 std::vector<MPI_Request>& receiveRequests)
119
120{
121 constexpr int keyTag = static_cast<int>(P2pTags::focusPeerCounts);
122 size_t numPeers = peerRanks.size();
123
124 std::vector<MPI_Request> sendRequests;
125 sendRequests.reserve(numPeers);
126 for (auto peer : peerRanks)
127 {
128 // +1 to include the upper key boundary for the last node
129 TreeNodeIndex sendCount = focusAssignment[peer].count() + 1;
130 mpiSendAsync(localLeaves.data() + focusAssignment[peer].start(), sendCount, peer, keyTag, sendRequests);
131 }
132
133 receiveRequests.reserve(numPeers);
134 int numMessages = numPeers;
135 while (numMessages > 0)
136 {
137 MPI_Status status;
138 MPI_Probe(MPI_ANY_SOURCE, keyTag, MPI_COMM_WORLD, &status);
139 int receiveRank = status.MPI_SOURCE;
140 TreeNodeIndex receiveSize;
141 MPI_Get_count(&status, MpiType<KeyType>{}, &receiveSize);
142 peerTrees[receiveRank].resize(receiveSize);
143
144 mpiRecvAsync(peerTrees[receiveRank].data(), receiveSize, receiveRank, keyTag, receiveRequests);
145
146 numMessages--;
147 }
148
149 MPI_Waitall(int(numPeers), sendRequests.data(), MPI_STATUS_IGNORE);
150}

152template<class T, class KeyType>
153void exchangeTreeletGeneral(gsl::span<const int> peerRanks,
154 const std::vector<std::vector<KeyType>>& peerTrees,
155 gsl::span<const IndexPair<TreeNodeIndex>> focusAssignment,
156 gsl::span<const KeyType> prefixes,
157 gsl::span<const TreeNodeIndex> levelRange,
158 gsl::span<const TreeNodeIndex> csToInternalMap,
159 gsl::span<T> quantities,
160 int commTag)
161{
162 size_t numPeers = peerRanks.size();
163 std::vector<std::vector<T>> sendBuffers;
164 sendBuffers.reserve(numPeers);
165
166 std::vector<MPI_Request> sendRequests;
167 sendRequests.reserve(numPeers);
168 for (auto peer : peerRanks)
169 {
170 TreeNodeIndex treeletSize = nNodes(peerTrees[peer]);
171 std::vector<T> buffer(treeletSize);
172 gsl::span<const KeyType> treelet(peerTrees[peer]);
173
174#pragma omp parallel for
175 for (TreeNodeIndex i = 0; i < treeletSize; ++i)
176 {
177 TreeNodeIndex internalIdx = locateNode(treelet[i], treelet[i + 1], prefixes.data(), levelRange.data());
178 assert(treelet[i] == decodePlaceholderBit(prefixes[internalIdx]));
179 buffer[i] = quantities[internalIdx];
180 }
181
182 mpiSendAsync(buffer.data(), treeletSize, peer, commTag, sendRequests);
183 sendBuffers.push_back(std::move(buffer));
184 }
185
186 std::vector<T> buffer;
187 for (auto peer : peerRanks)
188 {
189 TreeNodeIndex receiveCount = focusAssignment[peer].count();
190 buffer.resize(receiveCount);
191 mpiRecvSync(buffer.data(), receiveCount, peer, commTag, MPI_STATUS_IGNORE);
192
193 auto mapToInternal = csToInternalMap.subspan(focusAssignment[peer].start(), receiveCount);
194 scatter(mapToInternal, buffer.data(), quantities.data());
195 }
196
197 MPI_Waitall(int(sendRequests.size()), sendRequests.data(), MPI_STATUS_IGNORE);
198}

218template<class KeyType>
219void focusTransfer(gsl::span<const KeyType> cstree,
220 gsl::span<const unsigned> counts,
221 unsigned bucketSize,
222 int myRank,
223 KeyType oldFocusStart,
224 KeyType oldFocusEnd,
225 KeyType newFocusStart,
226 KeyType newFocusEnd,
227 std::vector<KeyType>& buffer)
228{
229 constexpr int ownerTag = static_cast<int>(P2pTags::focusTransfer);
230
231 std::vector<MPI_Request> sendRequests;
232 std::vector<std::vector<KeyType>> sendBuffers;
233
234 if (oldFocusStart < newFocusStart)
235 {
236 // current rank lost range [oldFocusStart : newFocusStart] to rank below
237 TreeNodeIndex start = findNodeAbove(cstree.data(), cstree.size(), oldFocusStart);
238 TreeNodeIndex end = findNodeAbove(cstree.data(), cstree.size(), newFocusStart);
239
240 size_t numNodes = end - start;
241 auto treelet = updateTreelet(gsl::span<const KeyType>(cstree.data() + start, numNodes + 1),
242 gsl::span<const unsigned>(counts.data() + start, numNodes), bucketSize);
243
244 mpiSendAsync(treelet.data(), int(treelet.size() - 1), myRank - 1, ownerTag, sendRequests);
245 sendBuffers.push_back(std::move(treelet));
246 }
247
248 if (newFocusEnd < oldFocusEnd)
249 {
250 // current rank lost range [newFocusEnd : oldFocusEnd] to rank above
251 TreeNodeIndex start = findNodeAbove(cstree.data(), cstree.size(), newFocusEnd);
252 TreeNodeIndex end = findNodeAbove(cstree.data(), cstree.size(), oldFocusEnd);
253
254 size_t numNodes = end - start;
255 auto treelet = updateTreelet(gsl::span<const KeyType>(cstree.data() + start, numNodes + 1),
256 gsl::span<const unsigned>(counts.data() + start, numNodes), bucketSize);
257
258 mpiSendAsync(treelet.data(), int(treelet.size() - 1), myRank + 1, ownerTag, sendRequests);
259 sendBuffers.push_back(std::move(treelet));
260 }
261
262 if (newFocusStart < oldFocusStart)
263 {
264 // current rank gained range [newFocusStart : oldFocusStart] from rank below
265 MPI_Status status;
266 MPI_Probe(myRank - 1, ownerTag, MPI_COMM_WORLD, &status);
267 TreeNodeIndex receiveSize;
268 MPI_Get_count(&status, MpiType<KeyType>{}, &receiveSize);
269
270 buffer.resize(buffer.size() + receiveSize);
271 mpiRecvSync(buffer.data() + buffer.size() - receiveSize, receiveSize, myRank - 1, ownerTag, &status);
272 }
273
274 if (oldFocusEnd < newFocusEnd)
275 {
276 // current rank gained range [oldFocusEnd : newFocusEnd] from rank above
277 MPI_Status status;
278 MPI_Probe(myRank + 1, ownerTag, MPI_COMM_WORLD, &status);
279 TreeNodeIndex receiveSize;
280 MPI_Get_count(&status, MpiType<KeyType>{}, &receiveSize);
281
282 buffer.resize(buffer.size() + receiveSize);
283 mpiRecvSync(buffer.data() + buffer.size() - receiveSize, receiveSize, myRank + 1, ownerTag, MPI_STATUS_IGNORE);
284 }
285
286 MPI_Waitall(int(sendRequests.size()), sendRequests.data(), MPI_STATUS_IGNORE);
287}
