54template<class KeyType, class T>
55class GlobalAssignment
56{
57public:
58 GlobalAssignment(int rank, int nRanks, unsigned bucketSize, const Box<T>& box = Box<T>{0, 1})
59 : myRank_(rank)
60 , numRanks_(nRanks)
61 , bucketSize_(bucketSize)
62 , box_(box)
63 {
64 unsigned level = log8ceil<KeyType>(100 * nRanks);
65 auto initialBoundaries = initialDomainSplits<KeyType>(nRanks, level);
66 std::vector<KeyType> init = computeSpanningTree<KeyType>(initialBoundaries);
67 tree_.update(init.data(), nNodes(init));
68 nodeCounts_ = std::vector<unsigned>(nNodes(init), bucketSize_ - 1);
69 }

83 template<class Reorderer, class Vector>
84 LocalIndex assign(BufferDescription bufDesc,
85 Reorderer& reorderFunctor,
86 Vector& /*sratch0*/,
87 Vector& /*scratch1*/,
88 KeyType* particleKeys,
89 const T* x,
90 const T* y,
91 const T* z)
92 {
93 // number of locally assigned particles to consider for global tree building
94 LocalIndex numParticles = bufDesc.end - bufDesc.start;
95
96 box_ = makeGlobalBox(x + bufDesc.start, y + bufDesc.start, z + bufDesc.start, numParticles, box_);
97
98 gsl::span<KeyType> keyView(particleKeys + bufDesc.start, numParticles);
99
100 // compute SFC particle keys only for particles participating in tree build
101 computeSfcKeys(x + bufDesc.start, y + bufDesc.start, z + bufDesc.start, sfcKindPointer(keyView.data()),
102 numParticles, box_);
103
104 // sort keys and keep track of ordering for later use
105 reorderFunctor.setMapFromCodes(keyView.begin(), keyView.end());
106
107 gsl::span<const KeyType> oldLeaves = tree_.treeLeaves();
108 std::vector<KeyType> oldBoundaries(assignment_.numRanks() + 1);
109 for (size_t rank = 0; rank < oldBoundaries.size() - 1; ++rank)
110 {
111 oldBoundaries[rank] = oldLeaves[assignment_.firstNodeIdx(rank)];
112 }
113 oldBoundaries.back() = nodeRange<KeyType>(0);
114
115 updateOctreeGlobal(keyView.begin(), keyView.end(), bucketSize_, tree_, nodeCounts_);
116
117 if (firstCall_)
118 {
119 firstCall_ = false;
120 while (!updateOctreeGlobal(keyView.begin(), keyView.end(), bucketSize_, tree_, nodeCounts_))
121 ;
122 }
123
124 auto newAssignment = singleRangeSfcSplit(nodeCounts_, numRanks_);
125 limitBoundaryShifts<KeyType>(oldBoundaries, tree_.treeLeaves(), nodeCounts_, newAssignment);
126 assignment_ = std::move(newAssignment);
127
128 exchanges_ =
129 createSendRanges<KeyType>(assignment_, tree_.treeLeaves(), {particleKeys + bufDesc.start, numParticles});
130
131 return domain_exchange::exchangeBufferSize(bufDesc, numPresent(), numAssigned());
132 }

153 template<class Reorderer, class Vector, class... Arrays>
154 auto distribute(BufferDescription bufDesc,
155 Reorderer& reorderFunctor,
156 Vector& /*scratch1*/,
157 Vector& /*scratch2*/,
158 KeyType* keys,
159 T* x,
160 T* y,
161 T* z,
162 Arrays... particleProperties) const
163 {
164 receiveLog_.clear();
165 exchangeParticles(exchanges_, myRank_, bufDesc, numAssigned(), reorderFunctor.getMap(), receiveLog_, x, y, z,
166 particleProperties...);
167
168 auto [newStart, newEnd] = domain_exchange::assignedEnvelope(bufDesc, numPresent(), numAssigned());
169 LocalIndex envelopeSize = newEnd - newStart;
170 gsl::span<KeyType> keyView(keys + newStart, envelopeSize);
171
172 computeSfcKeys(x + newStart, y + newStart, z + newStart, sfcKindPointer(keyView.begin()), envelopeSize, box_);
173 // sort keys and keep track of the ordering
174 reorderFunctor.setMapFromCodes(keyView.begin(), keyView.end());
175
176 return std::make_tuple(newStart, keyView.subspan(numSendDown(), numAssigned()));
177 }
