42template<class T>
43constexpr int determineAlignment(int n)
44{
45 if (sizeof(T) * n % 16 == 0) { return 16; }
46 else if (sizeof(T) * n % 8 == 0) { return 8; }
47 else { return alignof(T); }
48}

276template<int N, int I = 0>
277struct LexicographicalCompare
278{
279 template<class T, class F>
280 HOST_DEVICE_FUN constexpr static auto loop(const T* lhs, const T* rhs, F&& compare)
281 {
282 if (compare(lhs[I], rhs[I])) { return true; }
283 if (compare(rhs[I], lhs[I])) { return false; }
284 return LexicographicalCompare<N, I + 1>::loop(lhs, rhs, compare);
285 }
286};
