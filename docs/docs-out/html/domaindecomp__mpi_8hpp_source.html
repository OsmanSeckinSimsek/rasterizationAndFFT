44template<size_t Alignment, class... Arrays>
45size_t numElementsFit(size_t numBytesAvail, Arrays... arrays)
46{
47 constexpr int bytesPerElement = (... + sizeof(std::decay_t<decltype(*arrays)>));
48 numBytesAvail -= sizeof...(arrays) * Alignment;
49 return numBytesAvail / bytesPerElement;
50}

83template<class... Arrays>
84void exchangeParticles(const SendRanges& sends,
85 int thisRank,
86 BufferDescription bufDesc,
87 LocalIndex numParticlesAssigned,
88 const LocalIndex* ordering,
89 std::vector<std::tuple<int, LocalIndex>>& receiveLog,
90 Arrays... arrays)
91{
92 using TransferType = uint64_t;
93 constexpr int alignment = sizeof(TransferType);
94 constexpr int headerBytes = round_up(sizeof(uint64_t), alignment);
95 bool record = receiveLog.empty();
96 int domExTag = static_cast<int>(P2pTags::domainExchange) + (record ? 0 : 1);
97
98 std::vector<std::vector<char>> sendBuffers;
99 std::vector<MPI_Request> sendRequests;
100
101 for (int destinationRank = 0; destinationRank < sends.numRanks(); ++destinationRank)
102 {
103 size_t sendCount = sends.count(destinationRank);
104 if (destinationRank == thisRank || sendCount == 0) { continue; }
105
106 size_t numSent = 0;
107 while (numSent < sendCount)
108 {
109 size_t numRemaining = sendCount - numSent;
110 size_t numFit = numElementsFit<alignment>(INT_MAX * sizeof(TransferType) - headerBytes, arrays...);
111 size_t nextSendCount = std::min(numFit, numRemaining);
112
113 std::vector<char> sendBuffer(headerBytes + computeByteOffsets(nextSendCount, alignment, arrays...).back());
114 encodeSendCountCpu(nextSendCount, sendBuffer.data());
115 packArrays<alignment>(gatherCpu, ordering + sends[destinationRank] + numSent, nextSendCount,
116 sendBuffer.data() + headerBytes, arrays + bufDesc.start...);
117
118 mpiSendAsyncAs<TransferType>(sendBuffer.data(), sendBuffer.size(), destinationRank, domExTag, sendRequests);
119 numSent += nextSendCount;
120 sendBuffers.push_back(std::move(sendBuffer));
121 }
122 assert(numSent == sendCount);
123 }
124
125 LocalIndex numParticlesPresent = sends.count(thisRank);
126 LocalIndex receiveStart = domain_exchange::receiveStart(bufDesc, numParticlesPresent, numParticlesAssigned);
127 LocalIndex receiveEnd = receiveStart + numParticlesAssigned - numParticlesPresent;
128
129 std::vector<char> receiveBuffer;
130 while (receiveStart != receiveEnd)
131 {
132 MPI_Status status;
133 MPI_Probe(MPI_ANY_SOURCE, domExTag, MPI_COMM_WORLD, &status);
134 int receiveRank = status.MPI_SOURCE;
135 int receiveCountTransfer;
136 MPI_Get_count(&status, MpiType<TransferType>{}, &receiveCountTransfer);
137
138 receiveBuffer.resize(receiveCountTransfer * sizeof(TransferType));
139 mpiRecvSyncAs<TransferType>(receiveBuffer.data(), receiveBuffer.size(), receiveRank, domExTag, &status);
140
141 size_t receiveCount = decodeSendCountCpu(receiveBuffer.data());
142 assert(receiveStart + receiveCount <= receiveEnd);
143
144 LocalIndex receiveLocation = receiveStart;
145 if (record) { receiveLog.emplace_back(receiveRank, receiveStart); }
146 else { receiveLocation = domain_exchange::findInLog(receiveLog.begin(), receiveLog.end(), receiveRank); }
147
148 char* particleData = receiveBuffer.data() + headerBytes;
149 auto packTuple = packBufferPtrs<alignment>(particleData, receiveCount, (arrays + receiveLocation)...);
150 auto scatterRanges = [receiveCount](auto arrayPair) { std::copy_n(arrayPair[1], receiveCount, arrayPair[0]); };
151 util::for_each_tuple(scatterRanges, packTuple);
152
153 receiveStart += receiveCount;
154 }
155
156 if (not sendRequests.empty())
157 {
158 MPI_Status status[sendRequests.size()];
159 MPI_Waitall(int(sendRequests.size()), sendRequests.data(), status);
160 }
161
162 // If this process is going to send messages with rank/tag combinations
163 // already sent in this function, this can lead to messages being mixed up
164 // on the receiver side. This happens e.g. with repeated consecutive calls of
165 // this function.
166
167 // MUST call MPI_Barrier or any other collective MPI operation that enforces synchronization
168 // across all ranks before calling this function again.
169 // MPI_Barrier(MPI_COMM_WORLD);
170}
