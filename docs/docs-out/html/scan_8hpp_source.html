54template<class T1, class T2>
55void exclusiveScan(const T1* in, T2* out, size_t numElements)
56{
57 constexpr int blockSize = (8192 + 16384) / sizeof(T1);
58
59 int numThreads = 1;
60#ifdef _OPENMP
61#pragma omp parallel
62 {
63#pragma omp single
64 numThreads = omp_get_num_threads();
65 }
66#endif
67
68 T2 superBlock[2][numThreads + 1];
69 std::fill(superBlock[0], superBlock[0] + numThreads + 1, 0);
70 std::fill(superBlock[1], superBlock[1] + numThreads + 1, 0);
71
72 unsigned elementsPerStep = numThreads * blockSize;
73 unsigned nSteps = numElements / elementsPerStep;
74
75#pragma omp parallel num_threads(numThreads)
76 {
77 int tid = 0;
78#ifdef _OPENMP
79 tid = omp_get_thread_num();
80#endif
81 for (size_t step = 0; step < nSteps; ++step)
82 {
83 size_t stepOffset = step * elementsPerStep + tid * blockSize;
84
85 std::exclusive_scan(in + stepOffset, in + stepOffset + blockSize, out + stepOffset, 0);
86
87 superBlock[step % 2][tid] = out[stepOffset + blockSize - 1] + in[stepOffset + blockSize - 1];
88
89#pragma omp barrier
90
91 T2 tSum = superBlock[(step + 1) % 2][numThreads];
92 for (int t = 0; t < tid; ++t)
93 tSum += superBlock[step % 2][t];
94
95 if (tid == numThreads - 1) superBlock[step % 2][numThreads] = tSum + superBlock[step % 2][numThreads - 1];
96
97 std::for_each(out + stepOffset, out + stepOffset + blockSize, [shift = tSum](T2& val) { val += shift; });
98 }
99 }
100
101 // remainder
102 T2 stepSum = superBlock[(nSteps + 1) % 2][numThreads];
103 std::exclusive_scan(in + nSteps * elementsPerStep, in + numElements, out + nSteps * elementsPerStep, stepSum);
104}

106template<class T>
107T exclusiveScanSerialInplace(T* out, size_t num_elements, T init)
108{
109 T a = init;
110 T b = init;
111 for (size_t i = 0; i < num_elements; ++i)
112 {
113 a += out[i];
114 out[i] = b;
115 b = a;
116 }

123void exclusiveScan(T* out, size_t numElements)
124{

126
127 int numThreads = 1;

136 std::fill(superBlock[1], superBlock[1] + numThreads + 1, 0);

145 {
