40constexpr int clz32(uint32_t x)
41{
42 constexpr int debruijn32[32] = {0, 31, 9, 30, 3, 8, 13, 29, 2, 5, 7, 21, 12, 24, 28, 19,
43 1, 10, 4, 14, 6, 22, 25, 20, 11, 15, 23, 26, 16, 27, 17, 18};
44 x |= x >> 1u;
45 x |= x >> 2u;
46 x |= x >> 4u;
47 x |= x >> 8u;
48 x |= x >> 16u;
49 x++;
50 return debruijn32[x * 0x076be629 >> 27u];
51}

82constexpr int countLeadingZeros(uint32_t x)
83{
84#ifdef __CUDA_ARCH__
85 return __clz(x);
86 // with GCC and clang, we can use the builtin implementation
87 // this also works with the intel compiler, which also defines __GNUC__
88#elif defined(__GNUC__) || defined(__clang__)
89
90 // if the target architecture is Haswell or later,
91 // __builtin_clz(l) is implemented with the LZCNT instruction
92 // which returns the number of bits for an input of zero,
93 // so this check is not required in that case (flag: -march=haswell)
94 if (x == 0) return 8 * sizeof(uint32_t);
95 return __builtin_clz(x);
96
97#else
98 if (x == 0) return 8 * sizeof(uint64_t);
99 return detail::clz32(x);
100
101#endif
102}
