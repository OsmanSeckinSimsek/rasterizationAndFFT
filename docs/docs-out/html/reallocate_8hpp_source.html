42template<class Vector>
43void reallocateGeneric(Vector& vector, size_t size, double growthRate)
44{
45 size_t current_capacity = vector.capacity();
46
47 if (size > current_capacity)
48 {
49 size_t reserve_size = double(size) * growthRate;
50 vector.reserve(reserve_size);
51 }
52 vector.resize(size);
53}

68template<class Vector>
69void reallocateDestructive(Vector& vector, size_t size, double growthRate)
70{
71 if (size > vector.capacity())
72 {
73 // swap with an empty temporary to force deallocation
74 Vector().swap(vector);
75 }
76 reallocate(vector, size, growthRate);
77}

93template<class Vector>
94size_t reallocateBytes(Vector& vec, size_t numBytes)
95{
96 constexpr size_t elementSize = sizeof(typename Vector::value_type);
97 size_t originalSize = vec.size();
98
99 size_t currentSizeBytes = originalSize * elementSize;
100 if (currentSizeBytes < numBytes) { reallocateDestructive(vec, (numBytes + elementSize - 1) / elementSize, 1.01); }
101
102 return originalSize;
103}

106template<class... Vectors1, class... Vectors2>
107void lowMemReallocate(size_t size,
108 double growthFactor,
109 std::tuple<Vectors1&...> conserved,
110 std::tuple<Vectors2&...> scratch)
111{
112 // if the new size exceeds capacity, we first deallocate all scratch buffers to make space for the reallocations
113 util::for_each_tuple(
114 [size](auto& v)
115 {
116 if (size > v.capacity()) { std::decay_t<decltype(v)>{}.swap(v); }
117 },
118 scratch);
119 util::for_each_tuple([size, growthFactor](auto& v) { reallocate(v, size, growthFactor); }, conserved);
120 util::for_each_tuple([size, growthFactor](auto& v) { reallocate(v, size, growthFactor); }, scratch);
121}
