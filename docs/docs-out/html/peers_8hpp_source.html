62template<class T, template<class> class TreeType, class KeyType>
63std::vector<int> findPeersMac(int myRank,
64 const SpaceCurveAssignment& assignment,
65 const TreeType<KeyType>& domainTree,
66 const Box<T>& box,
67 float invThetaEff)
68{
69 KeyType domainStart = domainTree.codeStart(domainTree.toInternal(assignment.firstNodeIdx(myRank)));
70 KeyType domainEnd = domainTree.codeEnd(domainTree.toInternal(assignment.lastNodeIdx(myRank) - 1));
71
72 auto crossFocusPairs =
73 [domainStart, domainEnd, invThetaEff, &tree = domainTree, &box](TreeNodeIndex a, TreeNodeIndex b)
74 {
75 bool aFocusOverlap = overlapTwoRanges(domainStart, domainEnd, tree.codeStart(a), tree.codeEnd(a));
76 bool bInFocus = containedIn(tree.codeStart(b), tree.codeEnd(b), domainStart, domainEnd);
77 // node a has to overlap/be contained in the focus, while b must not be inside it
78 if (!aFocusOverlap || bInFocus) { return false; }
79
80 IBox aBox = sfcIBox(sfcKey(tree.codeStart(a)), tree.level(a));
81 IBox bBox = sfcIBox(sfcKey(tree.codeStart(b)), tree.level(b));
82 auto [aCenter, aSize] = centerAndSize<KeyType>(aBox, box);
83 auto [bCenter, bSize] = centerAndSize<KeyType>(bBox, box);
84 return !minVecMacMutual(aCenter, aSize, bCenter, bSize, box, invThetaEff);
85 };
86
87 auto m2l = [](TreeNodeIndex, TreeNodeIndex) {};
88
89 std::vector<int> peerRanks(assignment.numRanks(), 0);
90 auto p2p = [&domainTree, &assignment, &peerRanks](TreeNodeIndex /*a*/, TreeNodeIndex b)
91 {
92 int peerRank = assignment.findRank(domainTree.cstoneIndex(b));
93 if (peerRanks[peerRank] == 0) { peerRanks[peerRank] = 1; }
94 };
95
96 std::vector<KeyType> spanningNodeKeys(spanSfcRange(domainStart, domainEnd) + 1);
97 spanSfcRange(domainStart, domainEnd, spanningNodeKeys.data());
98 spanningNodeKeys.back() = domainEnd;
99
100 const KeyType* nodeKeys = domainTree.nodeKeys().data();
101 const TreeNodeIndex* levelRange = domainTree.levelRange().data();
102
103#pragma omp parallel for schedule(dynamic)
104 for (std::size_t i = 0; i < spanningNodeKeys.size() - 1; ++i)
105 {
106 TreeNodeIndex nodeIdx = locateNode(spanningNodeKeys[i], spanningNodeKeys[i + 1], nodeKeys, levelRange);
107 dualTraversal(domainTree, nodeIdx, 0, crossFocusPairs, m2l, p2p);
108 }
109
110 std::vector<int> ret;
111 for (int i = 0; i < int(peerRanks.size()); ++i)
112 {
113 if (peerRanks[i]) { ret.push_back(i); }
114 }
115
116 return ret;
117}

120template<template<class> class TreeType, class KeyType, class T>
121std::vector<int> findPeersMacStt(int myRank,
122 const SpaceCurveAssignment& assignment,
123 const TreeType<KeyType>& octree,
124 const Box<T>& box,
125 float invThetaEff)
126{
127 KeyType domainStart = octree.codeStart(octree.toInternal(assignment.firstNodeIdx(myRank)));
128 KeyType domainEnd = octree.codeEnd(octree.toInternal(assignment.lastNodeIdx(myRank) - 1));
129
130 std::vector<int> peers(assignment.numRanks());
131
132#pragma omp parallel for
133 for (TreeNodeIndex i = assignment.firstNodeIdx(myRank); i < assignment.lastNodeIdx(myRank); ++i)
134 {
135 TreeNodeIndex internalIdx = octree.toInternal(i);
136 IBox target = sfcIBox(sfcKey(octree.codeStart(internalIdx)), octree.level(internalIdx));
137 Vec3<T> targetCenter, targetSize;
138 std::tie(targetCenter, targetSize) = centerAndSize<KeyType>(target, box);
139
140 auto violatesMac =
141 [&targetCenter, &targetSize, &octree, &box, invThetaEff, domainStart, domainEnd](TreeNodeIndex idx)
142 {
143 KeyType nodeStart = octree.codeStart(idx);
144 KeyType nodeEnd = octree.codeEnd(idx);
145 // if the tree node with index idx is fully contained in the focus, we stop traversal
146 if (containedIn(nodeStart, nodeEnd, domainStart, domainEnd)) { return false; }
147
148 IBox sourceBox = sfcIBox(sfcKey(nodeStart), octree.level(idx));
149 auto [sourceCenter, sourceSize] = centerAndSize<KeyType>(sourceBox, box);
150 return !minVecMacMutual(targetCenter, targetSize, sourceCenter, sourceSize, box, invThetaEff);
151 };
152
153 auto markLeafIdx = [toLeaf = octree.toLeafOrder(), &peers, &assignment](TreeNodeIndex idx)
154 {
155 int peerRank = assignment.findRank(toLeaf[idx]);
156 peers[peerRank] = 1;
157 };
158
159 singleTraversal(octree.childOffsets().data(), violatesMac, markLeafIdx);
160 }
161
162 std::vector<int> ret;
163 for (int i = 0; i < int(peers.size()); ++i)
164 {
165 if (peers[i]) { ret.push_back(i); }
166 }
167
168 return ret;
169}
