62template<class... Arrays>
63auto computeByteOffsets(size_t count, int alignment, Arrays... arrays)
64{
65 static_assert((... && std::is_pointer_v<Arrays>)&&"all arrays must be pointers");
66
67 util::array<size_t, sizeof...(Arrays) + 1> byteOffsets{sizeof(std::decay_t<decltype(*arrays)>)...};
68 byteOffsets *= count;
69
71 for (size_t i = 0; i < byteOffsets.size(); ++i)
72 {
73 byteOffsets[i] = round_up(byteOffsets[i], alignment);
74 }
75
76 std::exclusive_scan(byteOffsets.begin(), byteOffsets.end(), byteOffsets.begin(), size_t(0));
77
78 return byteOffsets;
79}

81template<int alignment, class... Arrays>
82size_t computeTotalSendBytes(const SendList& sendList, int thisRank, size_t numBytesHeader, Arrays... arrays)
83{
84 size_t totalSendBytes = 0;
85 for (int destinationRank = 0; destinationRank < int(sendList.size()); ++destinationRank)
86 {
87 size_t sendCount = sendList[destinationRank].totalCount();
88 if (destinationRank == thisRank || sendCount == 0) { continue; }
89
90 size_t particleBytes = computeByteOffsets(sendCount, alignment, arrays...).back();
92 totalSendBytes += numBytesHeader + particleBytes;
93 }
94

144template<int alignment, class... Arrays>
145auto packBufferPtrs(char* packedBufferBase, size_t arraySize, Arrays... arrays)
146{
147 static_assert((... && std::is_pointer_v<Arrays>)&&"all arrays must be pointers");
148 constexpr int numArrays = sizeof...(Arrays);
149 constexpr util::array<size_t, numArrays> elementSizes{sizeof(std::decay_t<decltype(*arrays)>)...};
150 constexpr auto indices = util::makeIntegralTuple(std::make_index_sequence<numArrays>{});
151
152 const std::array<char*, numArrays> data{reinterpret_cast<char*>(arrays)...};
153
154 auto arrayByteOffsets = computeByteOffsets(arraySize, alignment, arrays...);
155
156 using Types = util::TypeList<util::array<float, sizeof(std::decay_t<decltype(*arrays)>) / sizeof(float)>...>;
157 using PtrTypes = util::Map<aux_traits::MakeL2ArrayPtrs, Types>;
158 using TupleType = util::Reduce<std::tuple, PtrTypes>;
159
160 TupleType ret;
161
162 auto packOneBuffer = [packedBufferBase, &data, &elementSizes, &arrayByteOffsets, &ret](auto index)
163 {
164 using ElementType = util::array<float, elementSizes[index] / sizeof(float)>;
165 auto* srcPtr = reinterpret_cast<ElementType*>(data[index]);
166 auto* packedPtr = reinterpret_cast<ElementType*>(packedBufferBase + arrayByteOffsets[index]);
167
168 std::get<index>(ret) = util::array<ElementType*, 2>{srcPtr, packedPtr};
169 };
170
171 util::for_each_tuple(packOneBuffer, indices);
172
173 return ret;
174}

177template<int alignment, class F, class... Arrays>
178std::size_t packArrays(F&& gather, const LocalIndex* ordering, LocalIndex numElements, char* buffer, Arrays... arrays)
179{
180 auto gatherArray = [&gather, numElements, ordering](auto arrayPair)
181 { gather(ordering, numElements, arrayPair[0], arrayPair[1]); };
182
183 auto packTuple = packBufferPtrs<alignment>(buffer, numElements, arrays...);
184 util::for_each_tuple(gatherArray, packTuple);
185
186 std::size_t numBytesPacked = computeByteOffsets(numElements, alignment, arrays...).back();
187 return numBytesPacked;
188}

194[[maybe_unused]] static LocalIndex
195exchangeBufferSize(BufferDescription bufDesc, LocalIndex numPresent, LocalIndex numAssigned)
196{
197 LocalIndex numIncoming = numAssigned - numPresent;
198
199 bool fitHead = bufDesc.start >= numIncoming;
200 bool fitTail = bufDesc.size - bufDesc.end >= numIncoming;
201
202 return (fitHead || fitTail) ? bufDesc.size : bufDesc.end + numIncoming;
203}

206[[maybe_unused]] static LocalIndex
207receiveStart(BufferDescription bufDesc, LocalIndex numPresent, LocalIndex numAssigned)
208{
209 LocalIndex numIncoming = numAssigned - numPresent;
210
211 bool fitHead = bufDesc.start >= numIncoming;
212 assert(fitHead || /*fitTail*/ bufDesc.size - bufDesc.end >= numIncoming);
213
214 if (fitHead) { return bufDesc.start - numIncoming; }
215 else { return bufDesc.end; }
216}

219[[maybe_unused]] static util::array<LocalIndex, 2>
220assignedEnvelope(BufferDescription bufDesc, LocalIndex numPresent, LocalIndex numAssigned)
221{
222 LocalIndex numIncoming = numAssigned - numPresent;
223
224 bool fitHead = bufDesc.start >= numIncoming;
225 assert(fitHead || /*fitTail*/ bufDesc.size - bufDesc.end >= numIncoming);
226
227 if (fitHead) { return {bufDesc.start - numIncoming, bufDesc.end}; }
228 else { return {bufDesc.start, bufDesc.end + numIncoming}; }

232LocalIndex findInLog(It first, It last, int rank)
233{
234 auto it = std::find_if(first, last, [rank](auto e) { return std::get<0>(e) == rank; });

239template<class Vector>
240void extractLocallyOwnedImpl(BufferDescription bufDesc,
241 LocalIndex numPresent,
242 LocalIndex numAssigned,
243 const LocalIndex* ordering,
244 Vector& buffer)
245{
246 Vector temp(numAssigned);
247
248 // extract what we already had before the exchange
249 gatherCpu(ordering, numPresent, buffer.data() + bufDesc.start, temp.data());
250
251 // extract what we received during the exchange
252 LocalIndex rStart = receiveStart(bufDesc, numPresent, numAssigned);
253 std::copy_n(buffer.data() + rStart, numAssigned - numPresent, temp.data() + numPresent);
254 swap(temp, buffer);
255}
