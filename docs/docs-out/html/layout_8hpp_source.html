72inline std::vector<IndexPair<TreeNodeIndex>>
73invertRanges(TreeNodeIndex first, gsl::span<const IndexPair<TreeNodeIndex>> ranges, TreeNodeIndex last)
74{
75 std::vector<IndexPair<TreeNodeIndex>> invertedRanges;
76
77 TreeNodeIndex currentIndex = first;
78 for (auto range : ranges)
79 {
80 if (range.start() == range.end()) { continue; }
81
82 assert(currentIndex <= range.start() && "non-empty ranges must be sorted\n");
83 if (currentIndex < range.start()) { invertedRanges.emplace_back(currentIndex, range.start()); }
84 currentIndex = range.end();
85 }
86 if (currentIndex < last) { invertedRanges.emplace_back(currentIndex, last); }
87
88 return invertedRanges;
89}

109template<class IntegralType>
110std::vector<IntegralType> extractMarkedElements(gsl::span<const IntegralType> source,
111 gsl::span<const int> flags,
112 TreeNodeIndex firstReqIdx,
113 TreeNodeIndex secondReqIdx)
114{
115 std::vector<IntegralType> requestKeys;
116
117 while (firstReqIdx != secondReqIdx)
118 {
119 // advance to first halo (or to secondReqIdx)
120 while (firstReqIdx < secondReqIdx && flags[firstReqIdx] == 0)
121 {
122 firstReqIdx++;
123 }
124
125 // add one request key range
126 if (firstReqIdx != secondReqIdx)
127 {
128 requestKeys.push_back(source[firstReqIdx]);
129 // advance until not a halo or end of range
130 while (firstReqIdx < secondReqIdx && flags[firstReqIdx] == 1)
131 {
132 firstReqIdx++;
133 }
134 requestKeys.push_back(source[firstReqIdx]);
135 }
136 }
137
138 return requestKeys;
139}

150inline void computeNodeLayout(gsl::span<const unsigned> focusLeafCounts,
151 gsl::span<const int> haloFlags,
152 TreeNodeIndex firstAssignedIdx,
153 TreeNodeIndex lastAssignedIdx,
154 gsl::span<LocalIndex> layout)
155{
156#pragma omp parallel for
157 for (TreeNodeIndex i = 0; i < TreeNodeIndex(focusLeafCounts.size()); ++i)
158 {
159 bool haveParticles = (firstAssignedIdx <= i && i < lastAssignedIdx) || haloFlags[i];
160 layout[i] = -int(haveParticles) & focusLeafCounts[i];
161 }
162
163 exclusiveScan(layout.data(), layout.size());
164}

175inline SendList computeHaloReceiveList(gsl::span<const LocalIndex> layout,
176 gsl::span<const int> haloFlags,
177 gsl::span<const TreeIndexPair> assignment,
178 gsl::span<const int> peerRanks)
179{
180 SendList ret(assignment.size());
181
182 for (int peer : peerRanks)
183 {
184 TreeNodeIndex peerStartIdx = assignment[peer].start();
185 TreeNodeIndex peerEndIdx = assignment[peer].end();
186
187 std::vector<LocalIndex> receiveRanges =
188 extractMarkedElements<LocalIndex>(layout, haloFlags, peerStartIdx, peerEndIdx);
189
190 for (std::size_t i = 0; i < receiveRanges.size(); i += 2)
191 {
192 ret[peer].addRange(receiveRanges[i], receiveRanges[i + 1]);
193 }
194 }
195
196 return ret;
197}

200template<class Gather, class... Arrays1, class... Arrays2>
201void gatherArrays(Gather&& gatherFunc,
202 const LocalIndex* ordering,
203 LocalIndex numElements,
204 LocalIndex inputOffset,
205 LocalIndex outputOffset,
206 std::tuple<Arrays1&...> arrays,
207 std::tuple<Arrays2&...> scratchBuffers)
208{
209 auto reorderArray = [ordering, numElements, inputOffset, outputOffset, &gatherFunc, &scratchBuffers](auto& array)
210 {
211 auto& swapSpace = util::pickType<decltype(array)>(scratchBuffers);
212 assert(swapSpace.size() == array.size());
213 gatherFunc(ordering, numElements, rawPtr(array) + inputOffset, rawPtr(swapSpace) + outputOffset);
214 swap(swapSpace, array);
215 };
216
217 util::for_each_tuple(reorderArray, arrays);
218}
