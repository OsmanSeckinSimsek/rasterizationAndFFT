51template<bool Pbc, class T, std::enable_if_t<Pbc, int> = 0>
52HOST_DEVICE_FUN constexpr T distanceSq(T x1, T y1, T z1, T x2, T y2, T z2, const Box<T>& box)
53{
54 bool pbcX = (box.boundaryX() == BoundaryType::periodic);
55 bool pbcY = (box.boundaryY() == BoundaryType::periodic);
56 bool pbcZ = (box.boundaryZ() == BoundaryType::periodic);
57
58 T dx = x1 - x2;
59 T dy = y1 - y2;
60 T dz = z1 - z2;
61 // this folds d into the periodic range [-l/2, l/2] for each dimension if enabled
62 dx -= pbcX * box.lx() * std::rint(dx * box.ilx());
63 dy -= pbcY * box.ly() * std::rint(dy * box.ily());
64 dz -= pbcZ * box.lz() * std::rint(dz * box.ilz());
65
66 return dx * dx + dy * dy + dz * dz;
67}

70template<bool Pbc, class T, std::enable_if_t<!Pbc, int> = 0>
71HOST_DEVICE_FUN constexpr T distanceSq(T x1, T y1, T z1, T x2, T y2, T z2, const Box<T>& /*box*/)
72{
73 T xx = x1 - x2;
74 T yy = y1 - y2;
75 T zz = z1 - z2;
76
77 return xx * xx + yy * yy + zz * zz;
78}

95template<class Tc, class Th, class KeyType>
96HOST_DEVICE_FUN unsigned findNeighbors(LocalIndex i,
97 const Tc* x,
98 const Tc* y,
99 const Tc* z,
100 const Th* h,
101 const OctreeNsView<Tc, KeyType>& tree,
102 const Box<Tc>& box,
103 unsigned ngmax,
104 LocalIndex* neighbors)
105{
106 auto xi = x[i];
107 auto yi = y[i];
108 auto zi = z[i];
109 auto hi = h[i];
110
111 auto radiusSq = Th(4.0) * hi * hi;
112 Vec3<Tc> particle{xi, yi, zi};
113 unsigned numNeighbors = 0;
114
115 auto pbc = BoundaryType::periodic;
116 bool anyPbc = box.boundaryX() == pbc || box.boundaryY() == pbc || box.boundaryZ() == pbc;
117 bool usePbc = anyPbc && !insideBox(particle, {Tc(2) * hi, Tc(2) * hi, Tc(2) * hi}, box);
118
119 auto overlapsPbc = [particle, radiusSq, centers = tree.centers, sizes = tree.sizes, &box](TreeNodeIndex idx)
120 {
121 auto nodeCenter = centers[idx];
122 auto nodeSize = sizes[idx];
123 return norm2(minDistance(particle, nodeCenter, nodeSize, box)) < radiusSq;
124 };
125
126 auto overlaps = [particle, radiusSq, centers = tree.centers, sizes = tree.sizes](TreeNodeIndex idx)
127 {
128 auto nodeCenter = centers[idx];
129 auto nodeSize = sizes[idx];
130 return norm2(minDistance(particle, nodeCenter, nodeSize)) < radiusSq;
131 };
132
133 auto searchBoxPbc =
134 [i, particle, radiusSq, &tree, x, y, z, ngmax, neighbors, &numNeighbors, &box](TreeNodeIndex idx)
135 {
136 TreeNodeIndex leafIdx = tree.internalToLeaf[idx];
137 LocalIndex firstParticle = tree.layout[leafIdx];
138 LocalIndex lastParticle = tree.layout[leafIdx + 1];
139
140 for (LocalIndex j = firstParticle; j < lastParticle; ++j)
141 {
142 if (j == i) { continue; }
143 if (distanceSq<true>(x[j], y[j], z[j], particle[0], particle[1], particle[2], box) < radiusSq)
144 {
145 if (numNeighbors < ngmax) { neighbors[numNeighbors] = j; }
146 numNeighbors++;
147 }
148 }
149 };
150
151 auto searchBox = [i, particle, radiusSq, &tree, x, y, z, ngmax, neighbors, &numNeighbors, &box](TreeNodeIndex idx)
152 {
153 TreeNodeIndex leafIdx = tree.internalToLeaf[idx];
154 LocalIndex firstParticle = tree.layout[leafIdx];
155 LocalIndex lastParticle = tree.layout[leafIdx + 1];
156
157 for (LocalIndex j = firstParticle; j < lastParticle; ++j)
158 {
159 if (j == i) { continue; }
160 if (distanceSq<false>(x[j], y[j], z[j], particle[0], particle[1], particle[2], box) < radiusSq)
161 {
162 if (numNeighbors < ngmax) { neighbors[numNeighbors] = j; }
163 numNeighbors++;
164 }
165 }
166 };
167
168 if (usePbc) { singleTraversal(tree.childOffsets, overlapsPbc, searchBoxPbc); }
169 else { singleTraversal(tree.childOffsets, overlaps, searchBox); }
170
171 return numNeighbors;
172}

174template<class T, class KeyType>
175void findNeighbors(const T* x,
176 const T* y,
177 const T* z,
178 const T* h,
179 LocalIndex firstId,
180 LocalIndex lastId,
181 const Box<T>& box,
182 const OctreeNsView<T, KeyType>& treeView,
183 unsigned ngmax,
184 LocalIndex* neighbors,
185 unsigned* neighborsCount)
186{
187 LocalIndex numWork = lastId - firstId;
188
189#pragma omp parallel for
190 for (LocalIndex i = 0; i < numWork; ++i)
191 {
192 LocalIndex id = i + firstId;
193 neighborsCount[i] = findNeighbors(id, x, y, z, h, treeView, box, ngmax, neighbors + i * ngmax);
194 }
195}
