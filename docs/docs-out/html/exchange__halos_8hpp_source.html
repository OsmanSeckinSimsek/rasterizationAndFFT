43template<class... Arrays>
44void haloexchange(int epoch, const SendList& incomingHalos, const SendList& outgoingHalos, Arrays... arrays)
45{
46 using IndexType = SendManifest::IndexType;
47 int haloExchangeTag = static_cast<int>(P2pTags::haloExchange) + epoch;
48
49 std::vector<std::vector<char>> sendBuffers;
50 std::vector<MPI_Request> sendRequests;
51
52 for (std::size_t destinationRank = 0; destinationRank < outgoingHalos.size(); ++destinationRank)
53 {
54 size_t sendCount = outgoingHalos[destinationRank].totalCount();
55 if (sendCount == 0) continue;
56
57 std::vector<char> buffer(computeByteOffsets(sendCount, 1, arrays...).back());
58
59 auto packSendBuffer = [outHalos = outgoingHalos[destinationRank]](auto arrayPair)
60 {
61 for (std::size_t rangeIdx = 0; rangeIdx < outHalos.nRanges(); ++rangeIdx)
62 {
63 std::copy_n(arrayPair[0] + outHalos.rangeStart(rangeIdx), outHalos.count(rangeIdx), arrayPair[1]);
64 arrayPair[1] += outHalos.count(rangeIdx);
65 }
66 };
67
68 auto packTuple = packBufferPtrs<1>(buffer.data(), sendCount, arrays...);
69 for_each_tuple(packSendBuffer, packTuple);
70
71 mpiSendAsync(buffer.data(), buffer.size(), destinationRank, haloExchangeTag, sendRequests);
72 sendBuffers.push_back(std::move(buffer));
73 }
74
75 int numMessages = 0;
76 std::size_t maxReceiveSize = 0;
77 for (const auto& incomingHalo : incomingHalos)
78 {
79 numMessages += int(incomingHalo.totalCount() > 0);
80 maxReceiveSize = std::max(maxReceiveSize, incomingHalo.totalCount());
81 }
82 size_t maxReceiveBytes = computeByteOffsets(maxReceiveSize, 1, arrays...).back();
83
84 std::vector<char> receiveBuffer(maxReceiveBytes);
85
86 while (numMessages > 0)
87 {
88 MPI_Status status;
89 mpiRecvSync(receiveBuffer.data(), receiveBuffer.size(), MPI_ANY_SOURCE, haloExchangeTag, &status);
90 int receiveRank = status.MPI_SOURCE;
91 size_t receiveCount = incomingHalos[receiveRank].totalCount();
92
93 auto scatterRanges = [inHalos = incomingHalos[receiveRank]](auto arrayPair)
94 {
95 for (size_t rangeIdx = 0; rangeIdx < inHalos.nRanges(); ++rangeIdx)
96 {
97 std::copy_n(arrayPair[1], inHalos.count(rangeIdx), arrayPair[0] + inHalos.rangeStart(rangeIdx));
98 arrayPair[1] += inHalos.count(rangeIdx);
99 }
100 };
101
102 auto packTuple = packBufferPtrs<1>(receiveBuffer.data(), receiveCount, arrays...);
103 for_each_tuple(scatterRanges, packTuple);
104
105 numMessages--;
106 }
107
108 if (not sendRequests.empty())
109 {
110 MPI_Status status[sendRequests.size()];
111 MPI_Waitall(int(sendRequests.size()), sendRequests.data(), status);
112 }
113
114 // MUST call MPI_Barrier or any other collective MPI operation that enforces synchronization
115 // across all ranks before calling this function again.
116 // MPI_Barrier(MPI_COMM_WORLD);
117}
