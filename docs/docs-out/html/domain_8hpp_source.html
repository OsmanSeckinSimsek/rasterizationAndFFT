196 template<class KeyVec, class VectorX, class VectorH, class... Vectors1, class... Vectors2>
197 void sync(KeyVec& particleKeys,
198 VectorX& x,
199 VectorX& y,
200 VectorX& z,
201 VectorH& h,
202 std::tuple<Vectors1&...> particleProperties,
203 std::tuple<Vectors2&...> scratchBuffers)
204 {
205 staticChecks<KeyVec, VectorX, VectorH, Vectors1...>(scratchBuffers);
206 auto& sfcOrder = std::get<sizeof...(Vectors2) - 1>(scratchBuffers);
207 ReorderFunctor_t<std::decay_t<decltype(sfcOrder)>> sorter(sfcOrder);
208
209 auto scratch = util::discardLastElement(scratchBuffers);
210
211 auto [exchangeStart, keyView] =
212 distribute(sorter, particleKeys, x, y, z, std::tuple_cat(std::tie(h), particleProperties), scratch);
213 // h is already reordered here for use in halo discovery
214 gatherArrays(sorter.gatherFunc(), sorter.getMap() + global_.numSendDown(), global_.numAssigned(), exchangeStart,
215 0, std::tie(h), util::reverse(scratch));
216
217 float invThetaEff = invThetaMinMac(theta_);
218 std::vector<int> peers = findPeersMac(myRank_, global_.assignment(), global_.octree(), box(), invThetaEff);
219
220 if (firstCall_)
221 {
222 focusTree_.converge(box(), keyView, peers, global_.assignment(), global_.treeLeaves(), global_.nodeCounts(),
223 invThetaEff, std::get<0>(scratch));
224 }
225 focusTree_.updateTree(peers, global_.assignment(), global_.treeLeaves());
226 focusTree_.updateCounts(keyView, global_.treeLeaves(), global_.nodeCounts(), std::get<0>(scratch));
227 focusTree_.updateMinMac(box(), global_.assignment(), global_.treeLeaves(), invThetaEff);
228
229 focusTree_.updateGeoCenters(box());
230
231 auto octreeView = focusTree_.octreeViewAcc();
232 const KeyType* focusLeaves = focusTree_.treeLeavesAcc().data();
233
234 reallocateDestructive(layout_, octreeView.numLeafNodes + 1, 1.01);
235 reallocateDestructive(layoutAcc_, octreeView.numLeafNodes + 1, 1.01);
236 halos_.discover(octreeView.prefixes, octreeView.childOffsets, octreeView.internalToLeaf, focusLeaves,
237 focusTree_.leafCountsAcc(), focusTree_.assignment(), {rawPtr(layoutAcc_), layoutAcc_.size()},
238 box(), rawPtr(h), std::get<0>(scratch));
239 halos_.computeLayout(focusTree_.treeLeaves(), focusTree_.leafCounts(), focusTree_.assignment(), peers, layout_);
240
241 updateLayout(sorter, exchangeStart, keyView, particleKeys, std::tie(h),
242 std::tuple_cat(std::tie(x, y, z), particleProperties), scratch);
243 setupHalos(particleKeys, x, y, z, h, scratch);
244 firstCall_ = false;
245 }

247 template<class KeyVec, class VectorX, class VectorH, class VectorM, class... Vectors1, class... Vectors2>
248 void syncGrav(KeyVec& particleKeys,
249 VectorX& x,
250 VectorX& y,
251 VectorX& z,
252 VectorH& h,
253 VectorM& m,
254 std::tuple<Vectors1&...> particleProperties,
255 std::tuple<Vectors2&...> scratchBuffers)
256 {
257 staticChecks<KeyVec, VectorX, VectorH, VectorM, Vectors1...>(scratchBuffers);
258 auto& sfcOrder = std::get<sizeof...(Vectors2) - 1>(scratchBuffers);
259 ReorderFunctor_t<std::decay_t<decltype(sfcOrder)>> sorter(sfcOrder);
260
261 auto scratch = util::discardLastElement(scratchBuffers);
262
263 auto [exchangeStart, keyView] =
264 distribute(sorter, particleKeys, x, y, z, std::tuple_cat(std::tie(h, m), particleProperties), scratch);
265 gatherArrays(sorter.gatherFunc(), sorter.getMap() + global_.numSendDown(), global_.numAssigned(), exchangeStart,
266 0, std::tie(x, y, z, h, m), util::reverse(scratch));
267
268 float invThetaEff = invThetaVecMac(theta_);
269 std::vector<int> peers = findPeersMac(myRank_, global_.assignment(), global_.octree(), box(), invThetaEff);
270
271 if (firstCall_)
272 {
273 // first rough convergence to avoid computing expansion centers of large nodes with a lot of particles
274 focusTree_.converge(box(), keyView, peers, global_.assignment(), global_.treeLeaves(), global_.nodeCounts(),
275 1.0, std::get<0>(scratch));
276
277 int converged = 0;
278 while (converged != numRanks_)
279 {
280 focusTree_.updateMinMac(box(), global_.assignment(), global_.treeLeaves(), invThetaEff);
281 converged = focusTree_.updateTree(peers, global_.assignment(), global_.treeLeaves());
282 focusTree_.updateCounts(keyView, global_.treeLeaves(), global_.nodeCounts(), std::get<0>(scratch));
283 focusTree_.updateCenters(rawPtr(x), rawPtr(y), rawPtr(z), rawPtr(m), global_.assignment(),
284 global_.octree(), box(), std::get<0>(scratch), std::get<1>(scratch));
285 focusTree_.updateMacs(box(), global_.assignment(), global_.treeLeaves());
286 MPI_Allreduce(MPI_IN_PLACE, &converged, 1, MPI_INT, MPI_SUM, MPI_COMM_WORLD);
287 }
288 }
289 focusTree_.updateMinMac(box(), global_.assignment(), global_.treeLeaves(), invThetaEff);
290 focusTree_.updateTree(peers, global_.assignment(), global_.treeLeaves());
291 focusTree_.updateCounts(keyView, global_.treeLeaves(), global_.nodeCounts(), std::get<0>(scratch));
292 focusTree_.updateCenters(rawPtr(x), rawPtr(y), rawPtr(z), rawPtr(m), global_.assignment(), global_.octree(),
293 box(), std::get<0>(scratch), std::get<1>(scratch));
294 focusTree_.updateMacs(box(), global_.assignment(), global_.treeLeaves());
295
296 focusTree_.updateGeoCenters(box());
297
298 auto octreeView = focusTree_.octreeViewAcc();
299 const KeyType* focusLeaves = focusTree_.treeLeavesAcc().data();
300
301 reallocateDestructive(layout_, octreeView.numLeafNodes + 1, 1.01);
302 reallocateDestructive(layoutAcc_, octreeView.numLeafNodes + 1, 1.01);
303 halos_.discover(octreeView.prefixes, octreeView.childOffsets, octreeView.internalToLeaf, focusLeaves,
304 focusTree_.leafCountsAcc(), focusTree_.assignment(), {rawPtr(layoutAcc_), layoutAcc_.size()},
305 box(), rawPtr(h), std::get<0>(scratch));
306 focusTree_.addMacs(halos_.haloFlags());
307 halos_.computeLayout(focusTree_.treeLeaves(), focusTree_.leafCounts(), focusTree_.assignment(), peers, layout_);
308
309 // diagnostics(keyView.size(), peers);
310
311 updateLayout(sorter, exchangeStart, keyView, particleKeys, std::tie(x, y, z, h, m), particleProperties,
312 scratch);
313 setupHalos(particleKeys, x, y, z, h, scratch);
314 firstCall_ = false;
315 }

324 template<class... Vectors, class SendBuffer, class ReceiveBuffer, class OVec>
325 void reapplySync(std::tuple<Vectors&...> arrays,
326 SendBuffer& sendBuffer,
327 ReceiveBuffer& receiveBuffer,
328 OVec& ordering) const
329 {
330 static_assert((... && !IsDeviceVector<Vectors>{}), "reapplySync only support for arrays on CPUs");
331 std::apply([this](auto&... arrays) { this->template checkSizesEqual(this->prevBufDesc_.size, arrays...); },
332 arrays);
333
334 LocalIndex exSize =
335 domain_exchange::exchangeBufferSize(prevBufDesc_, global_.numPresent(), global_.numAssigned());
336 lowMemReallocate(exSize, 1.01, arrays, {});
337
338 BufferDescription exDesc{prevBufDesc_.start, prevBufDesc_.end, exSize};
339 auto envelope = domain_exchange::assignedEnvelope(exDesc, global_.numPresent(), global_.numAssigned());
340 LocalIndex shift = prevBufDesc_.start - envelope[0];
341
342 // the intermediate, reconstructed ordering needed for the MPI particle exchange
343 std::vector<LocalIndex> prevOrd(prevBufDesc_.end - prevBufDesc_.start);
344 // the post-exchange ordering that was obtained by sorting after receiving the particles from domain exchange
345 std::vector<LocalIndex> orderingCpu;
346
347 auto* ord = (LocalIndex*)rawPtr(ordering);
348 if constexpr (HaveGpu<Accelerator>{})
349 {
350 static_assert(IsDeviceVector<OVec>{}, "Need ordering on GPU for GPU-accelerated domain");
351 orderingCpu.resize(envelope[1] - envelope[0]);
352 memcpyD2H((LocalIndex*)rawPtr(ordering), orderingCpu.size(), orderingCpu.data());
353 ord = orderingCpu.data();
354 }
355
356 std::transform(ord, ord + global_.numSendDown(), prevOrd.data(), [shift](auto i) { return i - shift; });
357 std::transform(ord + global_.numSendDown() + global_.numAssigned(), ord + envelope[1] - envelope[0],
358 prevOrd.data() + global_.numSendDown() + global_.numPresent(),
359 [shift](auto i) { return i - shift; });
360
361 std::apply([exDesc, o = prevOrd.data(), &sendBuffer, &receiveBuffer, this](auto&... a)
362 { global_.redoExchange(exDesc, o, sendBuffer, receiveBuffer, rawPtr(a)...); },
363 arrays);
364
365 lowMemReallocate(bufDesc_.size, 1.01, arrays, std::tie(sendBuffer, receiveBuffer));
366 gatherArrays(gatherCpu, ord + global_.numSendDown(), global_.numAssigned(), envelope[0], bufDesc_.start, arrays,
367 std::tie(sendBuffer, receiveBuffer));
368 }

438 template<class KeyVec, class... ConservedVectors, class ScratchBuffers>
439 void staticChecks(ScratchBuffers& scratchBuffers)
440 {
441 static_assert(std::is_same_v<typename KeyVec::value_type, KeyType>);
442 static_assert(std::tuple_size_v<ScratchBuffers> >= 3);
443
444 auto tup = util::discardLastElement(scratchBuffers);
445 constexpr auto indices = std::make_tuple(util::FindIndex<ConservedVectors&, std::decay_t<decltype(tup)>>{}...);
446
447 auto valueTypeCheck = [](auto index)
448 {
449 static_assert(
450 index < std::tuple_size_v<std::decay_t<decltype(tup)>>,
451 "one of the conserved fields has a value type that was not found among the available scratch buffers");
452 };
453 util::for_each_tuple(valueTypeCheck, indices);
454 }

456 template<class Sorter, class KeyVec, class VectorX, class... Vectors1, class... Vectors2>
457 auto distribute(Sorter& sorter,
458 KeyVec& keys,
459 VectorX& x,
460 VectorX& y,
461 VectorX& z,
462 std::tuple<Vectors1&...> particleProperties,
463 std::tuple<Vectors2&...> scratchBuffers)
464 {
465 initBounds(x.size());
466 auto distributedArrays = std::tuple_cat(std::tie(keys, x, y, z), particleProperties);
467 std::apply([size = x.size()](auto&... arrays) { checkSizesEqual(size, arrays...); }, distributedArrays);
468
469 // Global tree build and assignment
470 auto exchangeSize = global_.assign(bufDesc_, sorter, std::get<0>(scratchBuffers), std::get<1>(scratchBuffers),
471 rawPtr(keys), rawPtr(x), rawPtr(y), rawPtr(z));
472 lowMemReallocate(exchangeSize, 1.01, distributedArrays, scratchBuffers);
473
474 return std::apply(
475 [exchangeSize, &sorter, &scratchBuffers, this](auto&... arrays)
476 {
477 return global_.distribute({bufDesc_.start, bufDesc_.end, exchangeSize}, sorter,
478 std::get<0>(scratchBuffers), std::get<1>(scratchBuffers), rawPtr(arrays)...);
479 },
480 distributedArrays);
481 }

503 template<class Sorter, class KeyVec, class... Arrays1, class... Arrays2, class... Arrays3>
504 void updateLayout(Sorter& sorter,
505 LocalIndex exchangeStart,
506 gsl::span<const KeyType> keyView,
507 KeyVec& keys,
508 std::tuple<Arrays1&...> orderedBuffers,
509 std::tuple<Arrays2&...> unorderedBuffers,
510 std::tuple<Arrays3&...> scratchBuffers)
511 {
512 auto myRange = focusTree_.assignment()[myRank_];
513 BufferDescription newBufDesc{layout_[myRange.start()], layout_[myRange.end()], layout_.back()};
514
515 lowMemReallocate(newBufDesc.size, 1.01, std::tuple_cat(orderedBuffers, unorderedBuffers), scratchBuffers);
516
517 // re-locate particle SFC keys
518 if constexpr (IsDeviceVector<KeyVec>{})
519 {
520 memcpyH2D(layout_.data(), layout_.size(), rawPtr(layoutAcc_));
521 auto& swapSpace = std::get<0>(scratchBuffers);
522 size_t origSize = reallocateBytes(swapSpace, keyView.size() * sizeof(KeyType));
523
524 auto* swapPtr = reinterpret_cast<KeyType*>(rawPtr(swapSpace));
525 memcpyD2D(keyView.data(), keyView.size(), swapPtr);
526 reallocateDestructive(keys, newBufDesc.size, 1.01);
527 memcpyD2D(swapPtr, keyView.size(), rawPtr(keys) + newBufDesc.start);
528
529 reallocate(swapSpace, origSize, 1.0);
530 }
531 else
532 {
533 omp_copy(layout_.begin(), layout_.end(), layoutAcc_.begin());
534 reallocate(swapKeys_, newBufDesc.size, 1.01);
535 omp_copy(keyView.begin(), keyView.end(), swapKeys_.begin() + newBufDesc.start);
536 swap(keys, swapKeys_);
537 }
538
539 // relocate ordered buffer contents from offset 0 to offset newBufDesc.start
540 auto relocate =
541 [size = keyView.size(), dest = newBufDesc.start, scratch = util::reverse(scratchBuffers)](auto& array)
542 {
543 static_assert(util::FindIndex<decltype(array), std::tuple<Arrays3&...>>{} < sizeof...(Arrays3),
544 "No suitable scratch buffer available");
545 auto& swapSpace = util::pickType<decltype(array)>(scratch);
546 if constexpr (IsDeviceVector<std::decay_t<decltype(array)>>{})
547 {
548 memcpyD2D(rawPtr(array), size, rawPtr(swapSpace) + dest);
549 }
550 else { omp_copy(array.begin(), array.begin() + size, swapSpace.begin() + dest); }
551 swap(array, swapSpace);
552 };
553 util::for_each_tuple(relocate, orderedBuffers);
554
555 // reorder the unordered buffers
556 gatherArrays(sorter.gatherFunc(), sorter.getMap() + global_.numSendDown(), global_.numAssigned(), exchangeStart,
557 newBufDesc.start, unorderedBuffers, util::reverse(scratchBuffers));
558
559 // newBufDesc is now the valid buffer description
560 prevBufDesc_ = bufDesc_;
561 bufDesc_ = newBufDesc;
562 }

564 void diagnostics(size_t assignedSize, gsl::span<int> peers)
565 {
566 auto focusAssignment = focusTree_.assignment();
567 auto focusTree = focusTree_.treeLeaves();
568 auto globalTree = global_.treeLeaves();
569
570 TreeNodeIndex numFocusPeers = 0;
571 TreeNodeIndex numFocusTruePeer = 0;
572 for (int i = 0; i < numRanks_; ++i)
573 {
574 if (i != myRank_)
575 {
576 numFocusPeers += focusAssignment[i].count();
577 for (TreeNodeIndex fi = focusAssignment[i].start(); fi < focusAssignment[i].end(); ++fi)
578 {
579 KeyType fnstart = focusTree[fi];
580 KeyType fnend = focusTree[fi + 1];
581 TreeNodeIndex gi = findNodeAbove(globalTree, fnstart);
582 if (!(gi < nNodes(globalTree) && globalTree[gi] == fnstart && globalTree[gi + 1] <= fnend))
583 {
584 numFocusTruePeer++;
585 }
586 }
587 }
588 }
589
590 int numFlags = std::count(halos_.haloFlags().cbegin(), halos_.haloFlags().cend(), 1);
591 for (int i = 0; i < numRanks_; ++i)
592 {
593 if (i == myRank_)
594 {
595 std::cout << "rank " << i << " " << assignedSize << " " << layout_.back()
596 << " focus h/true/peers/loc/tot: " << numFlags << "/" << numFocusTruePeer << "/"
597 << numFocusPeers << "/" << focusAssignment[myRank_].count() << "/"
598 << halos_.haloFlags().size() << " peers: [" << peers.size() << "] ";
599 if (numRanks_ <= 32)
600 {
601 for (auto r : peers)
602 {
603 std::cout << r << " ";
604 }
605 }
606 std::cout << std::endl;
607 }
608 MPI_Barrier(MPI_COMM_WORLD);
609 }
610 }
