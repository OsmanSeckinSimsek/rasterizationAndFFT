93template<class KeyType>
94HOST_DEVICE_FUN unsigned calculateNodeCount(
95 KeyType nodeStart, KeyType nodeEnd, const KeyType* codesStart, const KeyType* codesEnd, size_t maxCount)
96{
97 // count particles in range
98 auto rangeStart = stl::lower_bound(codesStart, codesEnd, nodeStart);
99 auto rangeEnd = stl::lower_bound(codesStart, codesEnd, nodeEnd);
100 size_t count = rangeEnd - rangeStart;
101
102 return stl::min(count, maxCount);
103}

115template<class KeyType>
116HOST_DEVICE_FUN util::array<const KeyType*, 2> findSearchBounds(std::make_signed_t<KeyType> firstIdx,
117 KeyType targetCode,
118 const KeyType* codesStart,
119 const KeyType* codesEnd)
120{
121 assert(firstIdx >= 0);
122
123 using SI = std::make_signed_t<KeyType>;
124 SI nCodes = codesEnd - codesStart;
125
126 // firstIdx must be an accessible index
127 firstIdx = stl::min(nCodes - 1, firstIdx);
128
129 // the last node in 64-bit is 2^63, which can't be represented as a negative number
130 if (targetCode == nodeRange<KeyType>(0)) { return {codesStart + firstIdx, codesEnd}; }
131
132 KeyType firstCode = codesStart[firstIdx];
133 if (firstCode == targetCode) { firstIdx++; }
134
135 // d = 1 : search towards codesEnd
136 // d = -1 : search towards codesStart
137 SI d = (firstCode < targetCode) ? 1 : -1;
138
139 SI targetCodeTimesD = targetCode * d;
140
141 // determine search bound
142 SI searchRange = 1;
143 SI secondIndex = firstIdx + d;
144 while (0 <= secondIndex && secondIndex < nCodes && d * SI(codesStart[secondIndex]) <= targetCodeTimesD)
145 {
146 searchRange *= 2;
147 secondIndex = firstIdx + searchRange * d;
148 }
149 secondIndex = stl::max(SI(0), secondIndex);
150 secondIndex = stl::min(nCodes, secondIndex);
151
152 return {codesStart + stl::min(firstIdx, secondIndex), codesStart + stl::max(firstIdx, secondIndex)};
153}

168template<class KeyType>
169HOST_DEVICE_FUN unsigned updateNodeCount(TreeNodeIndex nodeIdx,
170 const KeyType* tree,
171 std::make_signed_t<KeyType> firstGuess,
172 std::make_signed_t<KeyType> secondGuess,
173 const KeyType* codesStart,
174 const KeyType* codesEnd,
175 unsigned maxCount)
176{
177 KeyType nodeStart = tree[nodeIdx];
178 KeyType nodeEnd = tree[nodeIdx + 1];
179
180 auto searchBounds = findSearchBounds(firstGuess, nodeStart, codesStart, codesEnd);
181 auto rangeStart = stl::lower_bound(searchBounds[0], searchBounds[1], nodeStart);
182
183 searchBounds = findSearchBounds(secondGuess, nodeEnd, codesStart, codesEnd);
184 auto rangeEnd = stl::lower_bound(searchBounds[0], searchBounds[1], nodeEnd);
185
186 unsigned count = rangeEnd - rangeStart;
187 return stl::min(count, maxCount);
188}

202template<class KeyType>
203void computeNodeCounts(const KeyType* tree,
204 unsigned* counts,
205 TreeNodeIndex nNodes,
206 const KeyType* codesStart,
207 const KeyType* codesEnd,
208 unsigned maxCount,
209 bool useCountsAsGuess = false)
210{
211 TreeNodeIndex firstNode = 0;
212 TreeNodeIndex lastNode = nNodes;
213 if (codesStart != codesEnd)
214 {
215 firstNode = std::upper_bound(tree, tree + nNodes, *codesStart) - tree - 1;
216 lastNode = std::upper_bound(tree, tree + nNodes, *(codesEnd - 1)) - tree;
217 assert(firstNode <= lastNode && "Are your particle codes sorted?");
218 }
219 else
220 {
221 firstNode = nNodes;
222 lastNode = nNodes;
223 }
224
225#pragma omp parallel for schedule(static)
226 for (TreeNodeIndex i = 0; i < firstNode; ++i)
227 counts[i] = 0;
228
229#pragma omp parallel for schedule(static)
230 for (TreeNodeIndex i = lastNode; i < nNodes; ++i)
231 counts[i] = 0;
232
233 TreeNodeIndex nNonZeroNodes = lastNode - firstNode;
234 const KeyType* populatedTree = tree + firstNode;
235
236 if (useCountsAsGuess)
237 {
238 exclusiveScan(counts + firstNode, nNonZeroNodes);
239#pragma omp parallel for schedule(static)
240 for (TreeNodeIndex i = 0; i < nNonZeroNodes; ++i)
241 {
242 unsigned firstGuess = counts[i + firstNode];
243 unsigned secondGuess = counts[std::min(i + firstNode + 1, nNodes - 1)];
244 counts[i + firstNode] =
245 updateNodeCount(i, populatedTree, firstGuess, secondGuess, codesStart, codesEnd, maxCount);
246 }
247 }
248 else
249 {
250#pragma omp parallel for schedule(static)
251 for (TreeNodeIndex i = 0; i < nNonZeroNodes; ++i)
252 {
253 counts[i + firstNode] =
254 calculateNodeCount(populatedTree[i], populatedTree[i + 1], codesStart, codesEnd, maxCount);
255 }
256 }
257}

272template<class KeyType>
273inline HOST_DEVICE_FUN util::tuple<int, unsigned> siblingAndLevel(const KeyType* csTree, TreeNodeIndex nodeIdx)
274{
275 KeyType thisNode = csTree[nodeIdx];
276 KeyType range = csTree[nodeIdx + 1] - thisNode;
277 unsigned level = treeLevel(range);
278
279 if (level == 0) { return {-1, level}; }
280
281 int siblingIdx = octalDigit(thisNode, level);
282 bool siblings = (csTree[nodeIdx - siblingIdx + 8] == csTree[nodeIdx - siblingIdx] + nodeRange<KeyType>(level - 1));
283 if (!siblings) { siblingIdx = -1; }
284
285 return {siblingIdx, level};
286}

289template<class KeyType>
290HOST_DEVICE_FUN int
291calculateNodeOp(const KeyType* tree, TreeNodeIndex nodeIdx, const unsigned* counts, unsigned bucketSize)
292{
293 auto [siblingIdx, level] = siblingAndLevel(tree, nodeIdx);
294
295 if (siblingIdx > 0) // 8 siblings next to each other, node can potentially be merged
296 {
297 // pointer to first node in sibling group
298 auto g = counts + nodeIdx - siblingIdx;
299#pragma GCC diagnostic ignored "-Warray-bounds"
300 size_t parentCount = size_t(g[0]) + size_t(g[1]) + size_t(g[2]) + size_t(g[3]) + size_t(g[4]) + size_t(g[5]) +
301 size_t(g[6]) + size_t(g[7]);
302 bool countMerge = parentCount <= size_t(bucketSize);
303 if (countMerge) { return 0; } // merge
304 }
305
306 if (counts[nodeIdx] > bucketSize * 512 && level + 3 < maxTreeLevel<KeyType>{}) { return 4096; } // split
307 if (counts[nodeIdx] > bucketSize * 64 && level + 2 < maxTreeLevel<KeyType>{}) { return 512; } // split
308 if (counts[nodeIdx] > bucketSize * 8 && level + 1 < maxTreeLevel<KeyType>{}) { return 64; } // split
309 if (counts[nodeIdx] > bucketSize && level < maxTreeLevel<KeyType>{}) { return 8; } // split
310
311 return 1; // default: do nothing
312}

331template<class KeyType, class LocalIndex>
332bool rebalanceDecision(
333 const KeyType* tree, const unsigned* counts, TreeNodeIndex nNodes, unsigned bucketSize, LocalIndex* nodeOps)
334{
335 bool converged = true;
336
337#pragma omp parallel
338 {
339 bool convergedThread = true;
340#pragma omp for
341 for (TreeNodeIndex i = 0; i < nNodes; ++i)
342 {
343 int decision = calculateNodeOp(tree, i, counts, bucketSize);
344 if (decision != 1) { convergedThread = false; }
345
346 nodeOps[i] = decision;
347 }
348 if (!convergedThread) { converged = false; }
349 }
350 return converged;
351}

361template<class KeyType>
362HOST_DEVICE_FUN void
363processNode(TreeNodeIndex nodeIndex, const KeyType* oldTree, const TreeNodeIndex* nodeOps, KeyType* newTree)
364{
365 KeyType thisNode = oldTree[nodeIndex];
366 KeyType range = oldTree[nodeIndex + 1] - thisNode;
367 unsigned level = treeLevel(range);
368
369 TreeNodeIndex opCode = nodeOps[nodeIndex + 1] - nodeOps[nodeIndex];
370 TreeNodeIndex newNodeIndex = nodeOps[nodeIndex];
371
372 if (opCode == 1) { newTree[newNodeIndex] = thisNode; }
373 else if (opCode == 8)
374 {
375 for (int sibling = 0; sibling < 8; ++sibling)
376 {
377 newTree[newNodeIndex + sibling] = thisNode + sibling * nodeRange<KeyType>(level + 1);
378 }
379 }
380 else if (opCode > 8)
381 {
382 unsigned levelDiff = log8ceil(unsigned(opCode));
383 for (int sibling = 0; sibling < opCode; ++sibling)
384 {
385 newTree[newNodeIndex + sibling] = thisNode + sibling * nodeRange<KeyType>(level + levelDiff);
386 }
387 }
388}

398template<class InputVector, class OutputVector>
399void rebalanceTree(const InputVector& tree, OutputVector& newTree, TreeNodeIndex* nodeOps)
400{
401 TreeNodeIndex numNodes = nNodes(tree);
402
403 exclusiveScan(nodeOps, numNodes + 1);
404 newTree.resize(nodeOps[numNodes] + 1);
405
406#pragma omp parallel for schedule(static)
407 for (TreeNodeIndex i = 0; i < numNodes; ++i)
408 {
409 processNode(i, tree.data(), nodeOps, newTree.data());
410 }
411 newTree.back() = tree.back();
412}

432template<class KeyType>
433bool updateOctree(const KeyType* codesStart,
434 const KeyType* codesEnd,
435 unsigned bucketSize,
436 std::vector<KeyType>& tree,
437 std::vector<unsigned>& counts,
438 unsigned maxCount = std::numeric_limits<unsigned>::max())
439{
440 std::vector<TreeNodeIndex> nodeOps(nNodes(tree) + 1);
441 bool converged = rebalanceDecision(tree.data(), counts.data(), nNodes(tree), bucketSize, nodeOps.data());
442
443 std::vector<KeyType> newTree;
444 rebalanceTree(tree, newTree, nodeOps.data());
445 swap(tree, newTree);
446
447 counts.resize(nNodes(tree));
448 computeNodeCounts(tree.data(), counts.data(), nNodes(tree), codesStart, codesEnd, maxCount, true);
449
450 return converged;
451}

454template<class KeyType>
455std::tuple<std::vector<KeyType>, std::vector<unsigned>>
456computeOctree(const KeyType* codesStart,
457 const KeyType* codesEnd,
458 unsigned bucketSize,
459 unsigned maxCount = std::numeric_limits<unsigned>::max())
460{
461 std::vector<KeyType> tree{0, nodeRange<KeyType>(0)};
462 std::vector<unsigned> counts{unsigned(codesEnd - codesStart)};
463
464 while (!updateOctree(codesStart, codesEnd, bucketSize, tree, counts, maxCount))
465 ;
466
467 return std::make_tuple(std::move(tree), std::move(counts));
468}

482updateTreelet(gsl::span<const KeyType> treelet, gsl::span<const unsigned> counts, unsigned bucketSize)
483{
484 std::vector<TreeNodeIndex> nodeOps(nNodes(treelet) + 1);

510template<class KeyType>
511std::vector<KeyType> computeSpanningTree(gsl::span<const KeyType> spanningKeys)
512{
513 assert(spanningKeys.size() > 1);
514 assert(spanningKeys.front() == 0 && spanningKeys.back() == nodeRange<KeyType>(0));
515
516 TreeNodeIndex numIntervals = spanningKeys.size() - 1;
517
518 std::vector<TreeNodeIndex> offsets(numIntervals + 1);
519 for (TreeNodeIndex i = 0; i < numIntervals; ++i)
520 {
521 offsets[i] = spanSfcRange(spanningKeys[i], spanningKeys[i + 1]);
522 }
523
524 exclusiveScanSerialInplace(offsets.data(), offsets.size(), 0);
525
526 std::vector<KeyType> spanningTree(offsets.back() + 1);
527 for (TreeNodeIndex i = 0; i < numIntervals; ++i)
528 {
529 spanSfcRange(spanningKeys[i], spanningKeys[i + 1], spanningTree.data() + offsets[i]);
530 }
531 spanningTree.back() = nodeRange<KeyType>(0);
532
533 return spanningTree;
534}
