73template<int R>
74HOST_DEVICE_FUN constexpr int pbcAdjust(int x)
75{
76 // this version handles x outside -R, 2R
77 // return x - R * std::floor(double(x)/R);
78 assert(x >= -R);
79 assert(x < 2 * R);
80 int ret = (x < 0) ? x + R : x;
81 return (ret >= R) ? ret - R : ret;
82}

85template<int R>
86HOST_DEVICE_FUN constexpr int pbcDistance(int x)
87{
88 // this version handles x outside -R, R
89 // int roundAwayFromZero = (x > 0) ? x + R/2 : x - R/2;
90 // return x -= R * (roundAwayFromZero / R);
91 assert(x >= -R);
92 assert(x <= R);
93 int ret = (x <= -R / 2) ? x + R : x;
94 return (ret > R / 2) ? ret - R : ret;
95}

194template<class T>
195HOST_DEVICE_FUN inline Vec3<T> applyPbc(Vec3<T> dX, const Box<T>& box)
196{
197 bool pbcX = (box.boundaryX() == BoundaryType::periodic);
198 bool pbcY = (box.boundaryY() == BoundaryType::periodic);
199 bool pbcZ = (box.boundaryZ() == BoundaryType::periodic);
200
201 dX[0] -= pbcX * box.lx() * std::rint(dX[0] * box.ilx());
202 dX[1] -= pbcY * box.ly() * std::rint(dX[1] * box.ily());
203 dX[2] -= pbcZ * box.lz() * std::rint(dX[2] * box.ilz());
204
205 return dX;
206}

209template<class T>
210HOST_DEVICE_FUN inline Vec3<T> putInBox(Vec3<T> X, const Box<T>& box)
211{
212 bool pbcX = (box.boundaryX() == BoundaryType::periodic);
213 bool pbcY = (box.boundaryY() == BoundaryType::periodic);
214 bool pbcZ = (box.boundaryZ() == BoundaryType::periodic);
215
216 // Further testing needed before this can be enabled
217 // X[0] -= pbcX * box.lx() * std::trunc(X[0] * box.ilx());
218 // X[1] -= pbcY * box.ly() * std::trunc(X[1] * box.ily());
219 // X[2] -= pbcZ * box.lz() * std::trunc(X[2] * box.ilz());
220
221 if (pbcX && X[0] > box.xmax()) { X[0] -= box.lx(); }
222 else if (pbcX && X[0] < box.xmin()) { X[0] += box.lx(); }
223
224 if (pbcY && X[1] > box.ymax()) { X[1] -= box.ly(); }
225 else if (pbcY && X[1] < box.ymin()) { X[1] += box.ly(); }
226
227 if (pbcZ && X[2] > box.zmax()) { X[2] -= box.lz(); }
228 else if (pbcZ && X[2] < box.zmin()) { X[2] += box.lz(); }
229
230 return X;
231}

234template<class Tc, class T>
235HOST_DEVICE_FUN inline void applyPBC(const cstone::Box<Tc>& box, T r, T& xx, T& yy, T& zz)
236{
237 bool pbcX = (box.boundaryX() == BoundaryType::periodic);
238 bool pbcY = (box.boundaryY() == BoundaryType::periodic);
239 bool pbcZ = (box.boundaryZ() == BoundaryType::periodic);
240
241 if (pbcX && xx > r)
242 xx -= box.lx();
243 else if (pbcX && xx < -r)
244 xx += box.lx();
245
246 if (pbcY && yy > r)
247 yy -= box.ly();
248 else if (pbcY && yy < -r)
249 yy += box.ly();
250
251 if (pbcZ && zz > r)
252 zz -= box.lz();
253 else if (pbcZ && zz < -r)
254 zz += box.lz();
255}

257template<class Tc, class T>
258HOST_DEVICE_FUN inline T distancePBC(const cstone::Box<Tc>& box, T hi, Tc x1, Tc y1, Tc z1, Tc x2, Tc y2, Tc z2)
259{
260 T xx = x1 - x2;
261 T yy = y1 - y2;
262 T zz = z1 - z2;
263
264 applyPBC(box, T(2) * hi, xx, yy, zz);
265
266 return std::sqrt(xx * xx + yy * yy + zz * zz);
267}

334template<class KeyType, class T>
335constexpr HOST_DEVICE_FUN util::tuple<Vec3<T>, Vec3<T>> centerAndSize(const IBox& ibox, const Box<T>& box)
336{
337 constexpr int maxCoord = 1u << maxTreeLevel<KeyType>{};
338 // smallest octree cell edge length in unit cube
339 constexpr T uL = T(1.) / maxCoord;
340
341 T halfUnitLengthX = T(0.5) * uL * box.lx();
342 T halfUnitLengthY = T(0.5) * uL * box.ly();
343 T halfUnitLengthZ = T(0.5) * uL * box.lz();
344 Vec3<T> boxCenter = {box.xmin() + (ibox.xmax() + ibox.xmin()) * halfUnitLengthX,
345 box.ymin() + (ibox.ymax() + ibox.ymin()) * halfUnitLengthY,
346 box.zmin() + (ibox.zmax() + ibox.zmin()) * halfUnitLengthZ};
347 Vec3<T> boxSize = {(ibox.xmax() - ibox.xmin()) * halfUnitLengthX, (ibox.ymax() - ibox.ymin()) * halfUnitLengthY,
348 (ibox.zmax() - ibox.zmin()) * halfUnitLengthZ};
349
350 return {boxCenter, boxSize};
351}

361template<class KeyType, class T>
362constexpr HOST_DEVICE_FUN FBox<T> createFpBox(const IBox& ibox, const Box<T>& box)
363{
364 auto [center, size] = centerAndSize<KeyType>(ibox, box);
365
366 auto Xmin = center - size;
367 auto Xmax = center + size;

381template<class KeyType, class T>
382HOST_DEVICE_FUN IBox createIBox(const Vec3<T> center, const Vec3<T>& size, const Box<T>& box)
383{
384 constexpr int maxCoord = 1u << maxTreeLevel<KeyType>{};
385
386 Vec3<T> Xmin = center - size;
387 Vec3<T> Xmax = center + size;
388
389 // normalize to units of box lengths
390 T xnMin = (Xmin[0] - box.xmin()) * box.ilx();
391 T ynMin = (Xmin[1] - box.ymin()) * box.ily();
392 T znMin = (Xmin[2] - box.zmin()) * box.ilz();
393
394 T xnMax = (Xmax[0] - box.xmin()) * box.ilx();
395 T ynMax = (Xmax[1] - box.ymin()) * box.ily();
396 T znMax = (Xmax[2] - box.zmin()) * box.ilz();
397
398 int ixMin = std::floor(xnMin * maxCoord);
399 int iyMin = std::floor(ynMin * maxCoord);
400 int izMin = std::floor(znMin * maxCoord);
401
402 int ixMax = std::ceil(xnMax * maxCoord);
403 int iyMax = std::ceil(ynMax * maxCoord);
404 int izMax = std::ceil(znMax * maxCoord);
405
406 return {ixMin, ixMax, iyMin, iyMax, izMin, izMax};
407}
