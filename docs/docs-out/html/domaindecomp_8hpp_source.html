57class SpaceCurveAssignment
58{
59 static constexpr TreeNodeIndex untouched = -1;
60
61public:
62 SpaceCurveAssignment()
63 : rankAssignment_(1)
64 {
65 }
66
67 explicit SpaceCurveAssignment(int numRanks)
68 : rankAssignment_(numRanks + 1, untouched)
69 , counts_(numRanks)
70 {
71 }
72
74 void addRange(int rank, TreeNodeIndex lower, TreeNodeIndex upper, std::size_t cnt)
75 {
76 // make sure that there's no holes or overlap between or with the range of the previous rank
77 assert(rankAssignment_[rank] == lower || rankAssignment_[rank] == untouched);
78
79 rankAssignment_[rank] = lower;
80 // will be overwritten by @p lower of rank+1, except if rank == numRanks-1
81 rankAssignment_[rank + 1] = upper;
82 counts_[rank] = cnt;
83 }
84
85 [[nodiscard]] int numRanks() const { return int(rankAssignment_.size()) - 1; }
86
87 [[nodiscard]] TreeNodeIndex firstNodeIdx(int rank) const { return rankAssignment_[rank]; }
88
89 [[nodiscard]] TreeNodeIndex lastNodeIdx(int rank) const { return rankAssignment_[rank + 1]; }
90
91 [[nodiscard]] int findRank(TreeNodeIndex nodeIdx) const
92 {
93 auto it = std::upper_bound(begin(rankAssignment_), end(rankAssignment_), nodeIdx);
94 return int(it - begin(rankAssignment_)) - 1;
95 }
96
98 [[nodiscard]] const std::size_t& totalCount(int rank) const { return counts_[rank]; }
99
100private:
101 friend bool operator==(const SpaceCurveAssignment& a, const SpaceCurveAssignment& b)
102 {
103 return a.rankAssignment_ == b.rankAssignment_ && a.counts_ == b.counts_;
104 }
105
106 std::vector<TreeNodeIndex> rankAssignment_;
107 std::vector<size_t> counts_;
108};

121inline SpaceCurveAssignment singleRangeSfcSplit(const std::vector<unsigned>& globalCounts, int nSplits)
122{
123 // one element per rank
124 SpaceCurveAssignment ret(nSplits);
125
126 std::size_t globalNParticles = std::accumulate(begin(globalCounts), end(globalCounts), std::size_t(0));
127
128 // distribute work, every rank gets global count / nSplits,
129 // the remainder gets distributed one by one
130 std::vector<std::size_t> nParticlesPerSplit(nSplits, globalNParticles / nSplits);
131 for (std::size_t split = 0; split < globalNParticles % nSplits; ++split)
132 {
133 nParticlesPerSplit[split]++;
134 }
135
136 TreeNodeIndex leavesDone = 0;
137 for (int split = 0; split < nSplits; ++split)
138 {
139 std::size_t targetCount = nParticlesPerSplit[split];
140 std::size_t splitCount = 0;
141 TreeNodeIndex j = leavesDone;
142 while (splitCount < targetCount && j < TreeNodeIndex(globalCounts.size()))
143 {
144 // if adding the particles of the next leaf takes us further away from
145 // the target count than where we're now, we stop
146 if (targetCount < splitCount + globalCounts[j] && // overshoot
147 targetCount - splitCount < splitCount + globalCounts[j] - targetCount) // overshoot more than undershoot
148 {
149 break;
150 }
151
152 splitCount += globalCounts[j++];
153 }
154
155 if (split < nSplits - 1)
156 {
157 // carry over difference of particles over/under assigned to next split
158 // to avoid accumulating round off
159 long int delta = (long int)(targetCount) - (long int)(splitCount);
160 nParticlesPerSplit[split + 1] += delta;
161 }
162 // afaict, j < nNodes(globalTree) can only happen if there are empty nodes at the end
163 else
164 {
165 for (; j < TreeNodeIndex(globalCounts.size()); ++j)
166 splitCount += globalCounts[j];
167 }
168
169 // other distribution strategies might have more than one range per rank
170 ret.addRange(split, leavesDone, j, splitCount);
171 leavesDone = j;
172 }
173
174 return ret;
175}

194template<class KeyType>
195void limitBoundaryShifts(gsl::span<const KeyType> oldBoundaries,
196 gsl::span<const KeyType> newTree,
197 gsl::span<const unsigned> counts,
198 SpaceCurveAssignment& newAssignment)
199{
200 // do nothing on first call when there are no boundaries
201 if (oldBoundaries.size() == 1) { return; }
202
203 int numRanks = newAssignment.numRanks();
204 std::vector<TreeNodeIndex> newIndexBoundaries(numRanks + 1, 0);
205 newIndexBoundaries.back() = newAssignment.lastNodeIdx(numRanks - 1);
206
207 bool triggerRecount = false;
208 for (int rank = 1; rank < numRanks; ++rank)
209 {
210 KeyType newBoundary = newTree[newAssignment.firstNodeIdx(rank)];
211
212 KeyType doNotGoBelow = oldBoundaries[rank - 1];
213 KeyType doNotExceed = oldBoundaries[rank + 1];
214
215 TreeNodeIndex restrictedStart = newAssignment.firstNodeIdx(rank);
216 if (newBoundary < doNotGoBelow)
217 {
218 restrictedStart = findNodeAbove(newTree.data(), newTree.size(), doNotGoBelow);
219 triggerRecount = true;
220 }
221 else if (newBoundary > doNotExceed)
222 {
223 restrictedStart = findNodeBelow(newTree.data(), newTree.size(), doNotExceed);
224 triggerRecount = true;
225 }
226 newIndexBoundaries[rank] = restrictedStart;
227 }
228
229 if (triggerRecount)
230 {
231 for (int rank = 0; rank < numRanks; ++rank)
232 {
233 std::size_t segmentCount = std::accumulate(counts.begin() + newIndexBoundaries[rank],
234 counts.begin() + newIndexBoundaries[rank + 1], std::size_t(0));
235 newAssignment.addRange(rank, newIndexBoundaries[rank], newIndexBoundaries[rank + 1], segmentCount);
236 }
237 }
238}

256template<class KeyType>
257void translateAssignment(const SpaceCurveAssignment& assignment,
258 gsl::span<const KeyType> domainTree,
259 gsl::span<const KeyType> focusTree,
260 gsl::span<const int> peerRanks,
261 int myRank,
262 std::vector<TreeIndexPair>& focusAssignment)
263{
264 focusAssignment.resize(assignment.numRanks());
265 std::fill(focusAssignment.begin(), focusAssignment.end(), TreeIndexPair(0, 0));
266 for (int peer : peerRanks)
267 {
268 KeyType peerSfcStart = domainTree[assignment.firstNodeIdx(peer)];
269 KeyType peerSfcEnd = domainTree[assignment.lastNodeIdx(peer)];
270
271 // Note: start-end range is narrowed down if no exact match is found.
272 // the discarded part will not participate in peer/halo exchanges
273 TreeNodeIndex startIndex = findNodeAbove(focusTree.data(), focusTree.size(), peerSfcStart);
274 TreeNodeIndex endIndex = findNodeBelow(focusTree.data(), focusTree.size(), peerSfcEnd);
275
276 if (endIndex < startIndex) { endIndex = startIndex; }
277 focusAssignment[peer] = TreeIndexPair(startIndex, endIndex);
278 }
279
280 KeyType startKey = domainTree[assignment.firstNodeIdx(myRank)];
281 KeyType endKey = domainTree[assignment.lastNodeIdx(myRank)];
282
283 TreeNodeIndex newStartIndex = findNodeAbove(focusTree.data(), focusTree.size(), startKey);
284 TreeNodeIndex newEndIndex = findNodeBelow(focusTree.data(), focusTree.size(), endKey);
285 focusAssignment[myRank] = TreeIndexPair(newStartIndex, newEndIndex);
286}

298template<class KeyType>
299SendRanges createSendRanges(const SpaceCurveAssignment& assignment,
300 gsl::span<const KeyType> treeLeaves,
301 gsl::span<const KeyType> particleKeys)
302{
303 int numRanks = assignment.numRanks();
304
305 SendRanges ret(numRanks + 1);
306 for (int rank = 0; rank < numRanks; ++rank)
307 {
308 KeyType rangeStart = treeLeaves[assignment.firstNodeIdx(rank)];
309 ret[rank] = std::lower_bound(particleKeys.begin(), particleKeys.end(), rangeStart) - particleKeys.begin();
310 }
311 ret.back() = particleKeys.size();
312
313 return ret;
314}

325template<class KeyType>
326std::vector<KeyType> initialDomainSplits(int numRanks, int level)
327{
328 std::vector<KeyType> ret(numRanks + 1);
329 KeyType delta = nodeRange<KeyType>(0) / numRanks;
330
331 ret.front() = 0;
332 for (int i = 1; i < numRanks; ++i)
333 {
334 ret[i] = enclosingBoxCode(KeyType(i) * delta, level);
335 }
336 ret.back() = nodeRange<KeyType>(0);
337
338 return ret;
339}
