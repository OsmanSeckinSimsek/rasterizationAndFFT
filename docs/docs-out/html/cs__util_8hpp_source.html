58template<class I>
59bool checkOctreeInvariants(const I* tree, int nNodes)
60{
61 // the root node delineated by code 0 and nodeRange<KeyType>(0)
62 // must be part of the tree
63 if (nNodes < 1) return false;
64 if (tree[0] != 0 || tree[nNodes] != nodeRange<I>(0)) return false;
65
66 for (int i = 0; i < nNodes; ++i)
67 {
68 if (i + 1 < nNodes && tree[i] >= tree[i + 1]) return false;
69
70 I range = tree[i + 1] - tree[i];
71
72 if (range == 0) return false;
73
74 if (!isPowerOf8(range)) return false;
75 }
76
77 return true;
78}

93template<class I>
94std::vector<I> makeNLevelGrid(int level)
95{
96 unsigned ticksPerDim = 1u << level;
97 I nNodes = ticksPerDim * ticksPerDim * ticksPerDim;
98 I tickRange = nodeRange<I>(level);
99
100 std::vector<I> codes;
101 codes.reserve(nNodes + 1);
102
103 for (unsigned i = 0; i < nNodes; ++i)
104 {
105 codes.push_back(i * tickRange);
106 }
107
108 return codes;
109}

135template<class I>
136inline I codeFromIndices(std::array<unsigned char, maxTreeLevel<uint64_t>{}> indices)
137{
138 constexpr unsigned nLevels = (sizeof(I) * 8) / 3;
139
140 I ret = 0;
141 for (unsigned idx = 0; idx < nLevels; ++idx)
142 {
143 assert(indices[idx] < 8);
144 unsigned treeLevel = nLevels - idx - 1;
145 ret += I(indices[idx]) << (3 * treeLevel);
146 }
147
148 return ret;
149}

171 OctreeMaker& divide(std::array<int, maxTreeLevel<uint64_t>{}> idx, std::size_t level)
172 {
173 std::array<unsigned char, maxTreeLevel<uint64_t>{}> indices{};
174 for (std::size_t i = 0; i < idx.size(); ++i)
175 indices[i] = static_cast<unsigned char>(idx[i]);
176
177 assert(std::find(begin(tree), end(tree), codeFromIndices<I>(indices)) != end(tree) &&
178 "node to be divided not present in tree");
179
180 indices[level] = 1;
181 assert(std::find(begin(tree), end(tree), codeFromIndices<I>(indices)) == end(tree) &&
182 "children of node to be divided already present in tree");
183
184 for (std::size_t sibling = 1; sibling < 8; ++sibling)
185 {
186 indices[level] = sibling;
187 tree.push_back(codeFromIndices<I>(indices));
188 }
189
190 return *this;
191 }
