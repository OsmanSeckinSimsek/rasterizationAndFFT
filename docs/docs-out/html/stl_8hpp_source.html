70{

81template<class ForwardIt, class T>
82HOST_DEVICE_FUN ForwardIt lower_bound(ForwardIt first, ForwardIt last, const T& value)
83{
84 ForwardIt it;
85 long long int step;
86 long long int count = last - first;
87
88 while (count > 0)
89 {
90 it = first;
91 step = count / 2;
92 it += step;
93 if (*it < value)
94 {
95 first = ++it;
96 count -= step + 1;
97 }
98 else
99 count = step;
100 }
101 return first;
102}
