58template<class T, class KeyType>
59HOST_DEVICE_FUN Vec4<T> computeMinMacR2(KeyType prefix, float invThetaEff, const Box<T>& box)
60{
61 KeyType nodeKey = decodePlaceholderBit(prefix);
62 int prefixLength = decodePrefixLength(prefix);
63
64 IBox cellBox = sfcIBox(sfcKey(nodeKey), prefixLength / 3);
65 auto [geoCenter, geoSize] = centerAndSize<KeyType>(cellBox, box);
66
67 T l = T(2) * max(geoSize);
68 T mac = l * invThetaEff;
69
70 return {geoCenter[0], geoCenter[1], geoCenter[2], mac * mac};
71}

81template<class T, class KeyType>
82HOST_DEVICE_FUN T computeVecMacR2(KeyType prefix, Vec3<T> expCenter, float invTheta, const Box<T>& box)
83{
84 KeyType nodeKey = decodePlaceholderBit(prefix);
85 int prefixLength = decodePrefixLength(prefix);
86
87 IBox cellBox = sfcIBox(sfcKey(nodeKey), prefixLength / 3);
88 auto [geoCenter, geoSize] = centerAndSize<KeyType>(cellBox, box);
89
90 Vec3<T> dX = expCenter - geoCenter;
91
92 T s = sqrt(norm2(dX));
93 T l = T(2.0) * max(geoSize);
94 T mac = l * invTheta + s;
95
96 return mac * mac;
97}

108template<class T>
109HOST_DEVICE_FUN bool evaluateMac(Vec3<T> sourceCenter, T macSq, Vec3<T> targetCenter, Vec3<T> targetSize)
110{
111 Vec3<T> dX = abs(targetCenter - sourceCenter) - targetSize;
112 dX += abs(dX);
113 dX *= T(0.5);
114 T R2 = norm2(dX);
115 return R2 < std::abs(macSq);
116}

129template<class T>
130HOST_DEVICE_FUN bool
131evaluateMacPbc(Vec3<T> sourceCenter, T macSq, Vec3<T> targetCenter, Vec3<T> targetSize, const Box<T>& box)
132{
133 Vec3<T> dX = targetCenter - sourceCenter;
134
135 dX = abs(applyPbc(dX, box));
136 dX -= targetSize;
137 dX += abs(dX);
138 dX *= T(0.5);
139 T R2 = norm2(dX);
140 return R2 < std::abs(macSq);
141}

144template<class T>
145HOST_DEVICE_FUN bool minMacMutual(const Vec3<T>& centerA,
146 const Vec3<T>& sizeA,
147 const Vec3<T>& centerB,
148 const Vec3<T>& sizeB,
149 const Box<T>& box,
150 float invTheta)
151{
152 Vec3<T> dX = minDistance(centerA, sizeA, centerB, sizeB, box);
153
154 T distSq = norm2(dX);
155 T sizeAB = 2 * stl::max(max(sizeA), max(sizeB));
156
157 T mac = sizeAB * invTheta;
158
159 return distSq > (mac * mac);
160}

170template<class T>
171HOST_DEVICE_FUN bool minVecMacMutual(const Vec3<T>& centerA,
172 const Vec3<T>& sizeA,
173 const Vec3<T>& centerB,
174 const Vec3<T>& sizeB,
175 const Box<T>& box,
176 float invThetaEff)
177{
178 bool passA;
179 {
180 // A = target, B = source
181 Vec3<T> dX = minDistance(centerB, centerA, sizeA, box);
182 T mac = max(sizeB) * 2 * invThetaEff;
183 passA = norm2(dX) > (mac * mac);
184 }
185 bool passB;
186 {
187 // B = target, A = source
188 Vec3<T> dX = minDistance(centerA, centerB, sizeB, box);
189 T mac = max(sizeA) * 2 * invThetaEff;
190 passB = norm2(dX) > (mac * mac);
191 }
192 return passA && passB;
193}

196template<class T, class KeyType>
197HOST_DEVICE_FUN void markMacPerBox(const Vec3<T>& targetCenter,
198 const Vec3<T>& targetSize,
199 const KeyType* prefixes,
200 const TreeNodeIndex* childOffsets,
201 const Vec4<T>* centers,
202 const Box<T>& box,
203 KeyType focusStart,
204 KeyType focusEnd,
205 char* markings)
206{
207 auto checkAndMarkMac =
208 [&targetCenter, &targetSize, prefixes, &box, focusStart, focusEnd, centers, markings](TreeNodeIndex idx)
209 {
210 KeyType nodePrefix = prefixes[idx];
211 KeyType nodeStart = decodePlaceholderBit(nodePrefix);
212 KeyType nodeEnd = nodeStart + nodeRange<KeyType>(decodePrefixLength(nodePrefix) / 3);
213 // if the tree node with index idx is fully contained in the focus, we stop traversal
214 if (containedIn(nodeStart, nodeEnd, focusStart, focusEnd)) { return false; }
215
216 Vec4<T> center = centers[idx];
217 bool violatesMac = evaluateMacPbc(makeVec3(center), center[3], targetCenter, targetSize, box);
218 if (violatesMac && !markings[idx]) { markings[idx] = 1; }
219
220 return violatesMac;
221 };
222
223 singleTraversal(childOffsets, checkAndMarkMac, [](TreeNodeIndex) {});
224}

239template<class T, class KeyType>
240void markMacs(const OctreeView<KeyType>& octree,
241 const Vec4<T>* centers,
242 const Box<T>& box,
243 KeyType focusStart,
244 KeyType focusEnd,
245 char* markings)
246
247{
248 std::fill(markings, markings + octree.numInternalNodes + octree.numLeafNodes, 0);
249
250 // find the minimum possible number of octree node boxes to cover the entire focus
251 TreeNodeIndex numFocusBoxes = spanSfcRange(focusStart, focusEnd);
252 std::vector<KeyType> focusCodes(numFocusBoxes + 1);
253 spanSfcRange(focusStart, focusEnd, focusCodes.data());
254 focusCodes.back() = focusEnd;
255
256#pragma omp parallel for schedule(dynamic)
257 for (TreeNodeIndex i = 0; i < numFocusBoxes; ++i)
258 {
259 IBox target = sfcIBox(sfcKey(focusCodes[i]), sfcKey(focusCodes[i + 1]));
260 auto [targetCenter, targetSize] = centerAndSize<KeyType>(target, box);
261 markMacPerBox(targetCenter, targetSize, octree.prefixes, octree.childOffsets, centers, box, focusStart,
262 focusEnd, markings);
263 }
264}
