49template<auto N,
50 int base,
51 typename char_type,
52 std::enable_if_t<std::is_integral_v<decltype(N)>, int> = 0,
53 std::enable_if_t<(base > 1 && base < sizeof(digits)), int> = 0>
54class to_string_t
55{
56public:
57 // The lambda calculates what the string length of N will be, so that `buf`
58 // fits to the number perfectly.
59 char_type buf[([]() constexpr noexcept {
60 unsigned int len = N > 0 ? 1 : 2;
61 for (auto n = N; n; len++, n /= base)
62 ;
63 return len;
64 }())] = {};
65
67 constexpr to_string_t() noexcept
68 {
69 auto ptr = end();
70 *--ptr = '\0';
71 if (N != 0)
72 {
73 for (auto n = N; n; n /= base)
74 *--ptr = digits[(N < 0 ? -1 : 1) * (n % base)];
75 if (N < 0) *--ptr = '-';
76 }
77 else { buf[0] = '0'; }
78 }
79
80 // Support implicit casting to `char *` or `const char *`.
81 constexpr operator char_type*() noexcept { return buf; }
82 constexpr operator const char_type*() const noexcept { return buf; }
83
84 constexpr auto size() const noexcept { return sizeof(buf) / sizeof(buf[0]); }
85 // Element access
86 constexpr auto data() noexcept { return buf; }
87 constexpr auto data() const noexcept { return buf; }
88 constexpr auto& operator[](unsigned int i) noexcept { return buf[i]; }
89 constexpr const auto& operator[](unsigned int i) const noexcept { return buf[i]; }
90 constexpr auto& front() noexcept { return buf[0]; }
91 constexpr const auto& front() const noexcept { return buf[0]; }
92 constexpr auto& back() noexcept { return buf[size() - 1]; }
93 constexpr const auto& back() const noexcept { return buf[size() - 1]; }
94 // Iterators
95 constexpr auto begin() noexcept { return buf; }
96 constexpr auto begin() const noexcept { return buf; }
97 constexpr auto end() noexcept { return buf + size(); }
98 constexpr auto end() const noexcept { return buf + size(); }
99};

116template<util::StructuralString Prefix, size_t... Is>
117constexpr std::array<const char*, sizeof...(Is)> enumerateFieldNames_helper(std::index_sequence<Is...>)
118{
119 constexpr size_t N = sizeof...(Is);
120 std::array<const char*, N> names;
121 [[maybe_unused]] std::initializer_list<int> list{
122 // CTAD within template arguments is broken in GCC 11.2 (https://gcc.gnu.org/bugzilla/show_bug.cgi?id=102933)
123 // therefore we must explicitly specify the template argument of StructuralString below
124 (names[Is] = ConstexprOnly<Prefix + StructuralString<to_string<Is>.size()>(to_string<Is>.buf)>, 0)...};
125 return names;
126}
