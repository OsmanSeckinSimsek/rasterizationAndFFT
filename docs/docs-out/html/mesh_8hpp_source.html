363 void fftfreq(std::vector<T>& freq, int n, double dt)
364 {
365 if (n % 2 == 0)
366 {
367 for (int i = 0; i < n / 2; i++)
368 {
369 freq[i] = i / (n * dt);
370 }
371 for (int i = n / 2; i < n; i++)
372 {
373 freq[i] = (i - n) / (n * dt);
374 }
375 }
376 else
377 {
378 for (int i = 0; i < (n - 1) / 2; i++)
379 {
380 freq[i] = i / (n * dt);
381 }
382 for (int i = (n - 1) / 2; i < n; i++)
383 {
384 freq[i] = (i - n + 1) / (n * dt);
385 }
386 }

389 // The normalized power spectrum results will be stored in rank 0
390 void perform_spherical_averaging(T* ps)
391 {
392 std::cout << "rank = " << rank_ << " spherical averaging started." << std::endl;
393 std::vector<T> k_values(gridDim_);
394 std::vector<T> k_1d(gridDim_);
395 std::vector<T> ps_rad(numShells_);
396 std::vector<int> count(k_1d.size());
397 std::vector<int> counts(k_1d.size(), 0);
398
399 fftfreq(k_values, gridDim_, 1.0 / gridDim_);
400
401#pragma omp parallel for
402 for (int i = 0; i < gridDim_; i++)
403 {
404 k_1d[i] = std::abs(k_values[i]);
405 }
406
407// iterate over the ps array and assign the values to the correct radial bin
408#pragma omp parallel for collapse(3)
409 for (int i = 0; i < inbox_.size[2]; i++) // slow heffte order
410 {
411 for (int j = 0; j < inbox_.size[1]; j++) // mid heffte order
412 {
413 for (int k = 0; k < inbox_.size[0]; k++) // fast heffte order
414 {
415 uint64_t freq_index = k + j * inbox_.size[0] + i * inbox_.size[0] * inbox_.size[1];
416
417 // Calculate the k indices with respect to the global mesh
418 uint64_t k_index_i = i + inbox_.low[2];
419 uint64_t k_index_j = j + inbox_.low[1];
420 uint64_t k_index_k = k + inbox_.low[0];
421 T kdist = std::sqrt(k_values[k_index_i] * k_values[k_index_i] +
422 k_values[k_index_j] * k_values[k_index_j] +
423 k_values[k_index_k] * k_values[k_index_k]);
424 std::vector<T> k_dif(gridDim_);
425
426 for (int kind = 0; kind < gridDim_; kind++)
427 {
428 k_dif[kind] = std::abs(k_1d[kind] - kdist);
429 }
430 auto it = std::min_element(std::begin(k_dif), std::end(k_dif));
431 uint64_t k_index = std::distance(std::begin(k_dif), it);
432
433#pragma omp atomic
434 ps_rad[k_index] += ps[freq_index];
435#pragma omp atomic
436 count[k_index]++;
437 }
438 }
439 }
440
441 MPI_Reduce(ps_rad.data(), power_spectrum_.data(), numShells_, MPI_DOUBLE, MPI_SUM, 0, MPI_COMM_WORLD);
442 MPI_Reduce(count.data(), counts.data(), numShells_, MPI_INT, MPI_SUM, 0, MPI_COMM_WORLD);
443
444 // normalize the power spectrum
445 if (rank_ == 0)
446 {
447 T sum_ps_radial = std::accumulate(power_spectrum_.begin(), power_spectrum_.end(), 0.0);
448
449 std::cout << "sum_ps_radial: " << sum_ps_radial << std::endl;
450
451#pragma omp parallel for
452 for (int i = 0; i < numShells_; i++)
453 {
454 if (count[i] != 0)
455 power_spectrum_[i] =
456 (power_spectrum_[i] * 4.0 * std::numbers::pi * std::pow(k_1d[i], 2)) / counts[i];
457 }
458 }
459 }
