118template<class I>
119HOST_DEVICE_FUN TreeNodeIndex findSplit(const I* sortedSfcCodes, TreeNodeIndex first, TreeNodeIndex last)
120{
121 // Identical Morton codes => split the range in the middle.
122 I firstCode = sortedSfcCodes[first];
123 I lastCode = sortedSfcCodes[last];
124
125 if (firstCode == lastCode) return (first + last) >> 1;
126
127 // Calculate the number of highest bits that are the same for all objects
128 int cpr = commonPrefix(firstCode, lastCode);
129
130 // Use binary search to find where the next bit differs.
131 // Specifically, we are looking for the highest Morton code that
132 // shares more than commonPrefix bits with the first one.
133 TreeNodeIndex split = first; // initial guess
134 TreeNodeIndex step = last - first;
135
136 do
137 {
138 step = (step + 1) / 2; // exponential decrease
139 TreeNodeIndex newSplit = split + step; // proposed new position
140
141 if (newSplit < last)
142 {
143 I splitCode = sortedSfcCodes[newSplit];
144 TreeNodeIndex splitPrefix = commonPrefix(firstCode, splitCode);
145 if (splitPrefix > cpr) split = newSplit; // accept proposal
146 }
147 } while (step > 1);
148
149 return split;
150}

161template<class I>
162HOST_DEVICE_FUN void
163constructInternalNode(const I* codes, TreeNodeIndex nCodes, BinaryNode<I>* internalNodes, TreeNodeIndex firstIndex)
164{
165 BinaryNode<I> outputNode;
166
167 int d = 1;
168 int minPrefixLength = -1;
169
170 if (firstIndex > 0)
171 {
172 d = (commonPrefix(codes[firstIndex], codes[firstIndex + 1]) >
173 commonPrefix(codes[firstIndex], codes[firstIndex - 1]))
174 ? 1
175 : -1;
176 minPrefixLength = commonPrefix(codes[firstIndex], codes[firstIndex - d]);
177 }
178
179 // determine searchRange, the maximum distance of secondIndex from firstIndex
180 TreeNodeIndex searchRange = 2;
181 TreeNodeIndex secondIndex = firstIndex + searchRange * d;
182 while (0 <= secondIndex && secondIndex < nCodes &&
183 commonPrefix(codes[firstIndex], codes[secondIndex]) > minPrefixLength)
184 {
185 searchRange *= 2;
186 secondIndex = firstIndex + searchRange * d;
187 }
188
189 // start binary search with known searchRange
190 secondIndex = firstIndex;
191 do
192 {
193 searchRange = (searchRange + 1) / 2;
194 TreeNodeIndex newJdx = secondIndex + searchRange * d;
195 if (0 <= newJdx && newJdx < nCodes && commonPrefix(codes[firstIndex], codes[newJdx]) > minPrefixLength)
196 {
197 secondIndex = newJdx;
198 }
199 } while (searchRange > 1);
200
201 int prefixLength = commonPrefix(codes[firstIndex], codes[secondIndex]);
202 I prefix = zeroLowBits(codes[firstIndex], prefixLength);
203 outputNode.prefix = encodePlaceholderBit(prefix, prefixLength);
204
205 // find position of highest differing bit between [firstIndex, secondIndex]
206 TreeNodeIndex gamma = findSplit(codes, stl::min(secondIndex, firstIndex), stl::max(secondIndex, firstIndex));
207
208 // establish child relationships
209 if (stl::min(secondIndex, firstIndex) == gamma)
210 {
211 // left child is a leaf
212 outputNode.child[BinaryNode<I>::left] = storeLeafIndex(gamma);
213 }
214 else
215 {
216 // left child is an internal binary node
217 outputNode.child[BinaryNode<I>::left] = gamma;
218 }
219
220 if (stl::max(secondIndex, firstIndex) == gamma + 1)
221 {
222 // right child is a leaf
223 outputNode.child[BinaryNode<I>::right] = storeLeafIndex(gamma + 1);
224 }
225 else
226 {
227 // right child is an internal binary node
228 outputNode.child[BinaryNode<I>::right] = gamma + 1;
229 }
230
231 internalNodes[firstIndex] = outputNode;
232}
