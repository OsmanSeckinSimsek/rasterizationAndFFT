75template<class Index>
76class IndexRanges
77{
78public:
79 using IndexType = Index;
80
81 IndexRanges()
82 : totalCount_(0)
83 , scan_{0}
84 {
85 }
86
88 void addRange(IndexType lower, IndexType upper)
89 {
90 assert(lower <= upper);
91 if (lower == upper) { return; }
92
93 totalCount_ += upper - lower;
94 offsets_.push_back(lower);
95 scan_.push_back(totalCount_);
96 }
97
98 [[nodiscard]] IndexType rangeStart(size_t i) const { return offsets_[i]; }
99
100 [[nodiscard]] IndexType rangeEnd(size_t i) const { return rangeStart(i) + count(i); }
101
103 [[nodiscard]] std::size_t count(size_t i) const { return scan_[i + 1] - scan_[i]; }
104
106 [[nodiscard]] std::size_t totalCount() const { return totalCount_; }
107
108 [[nodiscard]] std::size_t nRanges() const { return offsets_.size(); }
109
111 const IndexType* offsets() const { return offsets_.data(); }
112 const IndexType* scan() const { return scan_.data(); }
113
114private:
115 friend bool operator==(const IndexRanges& lhs, const IndexRanges& rhs)
116 {
117 return lhs.totalCount_ == rhs.totalCount_ && lhs.offsets_ == rhs.offsets_ && lhs.scan_ == rhs.scan_;
118 }
119
120 std::size_t totalCount_;
121 std::vector<IndexType> offsets_;
122 std::vector<IndexType> scan_;
123};
