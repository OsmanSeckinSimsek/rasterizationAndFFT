156template<class KeyType, class T>
157HOST_DEVICE_FUN inline KeyType sfc3D(T x, T y, T z, T xmin, T ymin, T zmin, T mx, T my, T mz)
158{
159 constexpr int mcoord = (1u << maxTreeLevel<typename KeyType::ValueType>{}) - 1;
160
161 int ix = std::floor(x * mx) - xmin * mx;
162 int iy = std::floor(y * my) - ymin * my;
163 int iz = std::floor(z * mz) - zmin * mz;
164
165 ix = stl::min(ix, mcoord);
166 iy = stl::min(iy, mcoord);
167 iz = stl::min(iz, mcoord);
168
169 assert(ix >= 0);
170 assert(iy >= 0);
171 assert(iz >= 0);
172
173 return iSfcKey<KeyType>(ix, iy, iz);
174}

233template<class KeyType, class T>
234HOST_DEVICE_FUN inline KeyType commonNodePrefix(Vec3<T> center, Vec3<T> size, const cstone::Box<T>& box)
235{
236 KeyType lowerKey = cstone::sfc3D<KeyType>(center[0] - size[0], center[1] - size[1], center[2] - size[2], box);
237 KeyType upperKey = cstone::sfc3D<KeyType>(center[0] + size[0], center[1] + size[1], center[2] + size[2], box);
238
239 unsigned level = commonPrefix(lowerKey, upperKey) / 3;
240 KeyType nodeKey = enclosingBoxCode(lowerKey, level);
241
242 return KeyType(encodePlaceholderBit(nodeKey.value(), 3 * level));
243}

254template<class KeyType>
255HOST_DEVICE_FUN inline KeyType sfcNeighbor(const IBox& ibox, unsigned level, int dx, int dy, int dz)
256{
257 constexpr unsigned pbcRange = 1u << maxTreeLevel<KeyType>{};
258
259 unsigned shiftValue = ibox.xmax() - ibox.xmin();
260
261 // lower corner of shifted box
262 int x = pbcAdjust<pbcRange>(ibox.xmin() + dx * shiftValue);
263 int y = pbcAdjust<pbcRange>(ibox.ymin() + dy * shiftValue);
264 int z = pbcAdjust<pbcRange>(ibox.zmin() + dz * shiftValue);
265
266 KeyType key = iSfcKey<KeyType>(x, y, z);
267
268 return KeyType(enclosingBoxCode(key, level));
269}
