41template<class KeyType, class T, class F>
42HOST_DEVICE_FUN unsigned continuumCount(KeyType nodeStart, KeyType nodeEnd, const Box<T>& box, F&& concentration)
43{
44 IBox nodeBox = sfcIBox(sfcKey(nodeStart), sfcKey(nodeEnd));
45 auto [center, size] = centerAndSize<KeyType>(nodeBox, box);
46
47 T volume = size[0] * size[1] * size[2];
48
49 double count = 0;
50 for (int ix = -1; ix <= 1; ix += 2)
51 for (int iy = -1; iy <= 1; iy += 2)
52 for (int iz = -1; iz <= 1; iz += 2)
53 {
54 auto corner = center + 0.5 * Vec3<T>{ix * size[0], iy * size[1], iz * size[2]};
55 T c = concentration(corner[0], corner[1], corner[2]);
56 count += c * volume;
57 }
58
59 return stl::min(unsigned(std::round(count)), std::numeric_limits<unsigned>::max());
60}

63void computeContinuumCounts(
64 const KeyType* tree, unsigned* counts, TreeNodeIndex numNodes, const Box<T>& box, F&& concentration)
65{
66#pragma ompe parallel for schedule(static)
67 for (TreeNodeIndex i = 0; i < numNodes; ++i)
68 {
69 counts[i] = continuumCount(tree[i], tree[i + 1], box, concentration);

73template<class F, class T, class KeyType>
74bool updateContinuumCsarray(F&& concentration,
75 const Box<T>& box,
76 unsigned bucketSize,
77 std::vector<KeyType>& tree,
78 std::vector<unsigned>& counts)
79{
80 std::vector<TreeNodeIndex> nodeOps(nNodes(tree) + 1);
81 bool converged = rebalanceDecision(tree.data(), counts.data(), nNodes(tree), bucketSize, nodeOps.data());
82
83 std::vector<KeyType> newTree;
84 rebalanceTree(tree, newTree, nodeOps.data());
85 swap(tree, newTree);
86
87 counts.resize(nNodes(tree));
88 computeContinuumCounts(tree.data(), counts.data(), nNodes(tree), box, concentration);
89
90 return converged;
91}

103template<class KeyType, class F, class T>
104std::tuple<std::vector<KeyType>, std::vector<unsigned>>
105computeContinuumCsarray(F&& concentration, const Box<T>& box, unsigned bucketSize)
106{
107 std::vector<KeyType> tree{0, nodeRange<KeyType>(0)};
108 std::vector<unsigned> counts{bucketSize + 1};
109
110 int maxIteration = 10;
111 while (!updateContinuumCsarray(concentration, box, bucketSize, tree, counts) && maxIteration--)
112 ;

114 return std::make_tuple(std::move(tree), std::move(counts));
