57void checkIndices(const SendList& sendList,
58 [[maybe_unused]] LocalIndex start,
59 [[maybe_unused]] LocalIndex end,
60 [[maybe_unused]] LocalIndex bufferSize)
61{
62 for (const auto& manifest : sendList)
63 {
64 for (size_t ri = 0; ri < manifest.nRanges(); ++ri)
65 {
66 assert(!overlapTwoRanges(LocalIndex{0}, start, manifest.rangeStart(ri), manifest.rangeEnd(ri)));
67 assert(!overlapTwoRanges(end, bufferSize, manifest.rangeStart(ri), manifest.rangeEnd(ri)));
68 }
69 }
70}

73void checkHalos(int myRank, gsl::span<const TreeIndexPair> focusAssignment, gsl::span<const int> haloFlags)
74{
75 TreeNodeIndex firstAssignedNode = focusAssignment[myRank].start();
76 TreeNodeIndex lastAssignedNode = focusAssignment[myRank].end();
77
78 std::array<TreeNodeIndex, 2> checkRanges[2] = {{0, firstAssignedNode},
79 {lastAssignedNode, TreeNodeIndex(haloFlags.size())}};
80
81 for (int range = 0; range < 2; ++range)
82 {
83#pragma omp parallel for
84 for (TreeNodeIndex i = checkRanges[range][0]; i < checkRanges[range][1]; ++i)
85 {
86 if (haloFlags[i])
87 {
88 bool peerFound = false;
89 for (auto peerRange : focusAssignment)
90 {
91 if (peerRange.start() <= i && i < peerRange.end()) { peerFound = true; }
92 }
93 if (!peerFound)
94 {
95 std::cout << "Detected halo cells not belonging to peer ranks. This usually happens"
96 << " when some particles have smoothing length interaction sphere volumes"
97 << " of similar magnitude than the rank domain volume. In that case, either"
98 << " the number of ranks needs to be decreased or the number of particles"
99 << " increased, leading to shorter smoothing lengths\n";
100 MPI_Abort(MPI_COMM_WORLD, 35);
101 }
102 }
103 }
104 }
105}

136 template<class T, class Th, class Vector>
137 void discover(const KeyType* prefixes,
138 const TreeNodeIndex* childOffsets,
139 const TreeNodeIndex* internalToLeaf,
140 const KeyType* leaves,
141 gsl::span<const unsigned> counts,
142 gsl::span<const TreeIndexPair> focusAssignment,
143 gsl::span<LocalIndex> layout,
144 const Box<T>& box,
145 const Th* h,
146 Vector& scratch)
147 {
148 TreeNodeIndex firstNode = focusAssignment[myRank_].start();
149 TreeNodeIndex lastNode = focusAssignment[myRank_].end();
150 TreeNodeIndex numNodesSearch = lastNode - firstNode;
151 TreeNodeIndex numLeafNodes = counts.size();
152
153 reallocate(numLeafNodes, haloFlags_);
154
155 if constexpr (HaveGpu<Accelerator>{})
156 {
157 // round up to multiple of 128 such that the radii pointer will be aligned
158 size_t flagBytes = round_up((numLeafNodes + 1) * sizeof(int), 128);
159 size_t radiiBytes = numLeafNodes * sizeof(float);
160 size_t origSize = reallocateBytes(scratch, flagBytes + radiiBytes);
161
162 auto* d_flags = reinterpret_cast<int*>(rawPtr(scratch));
163 auto* d_radii = reinterpret_cast<float*>(rawPtr(scratch)) + flagBytes / sizeof(float);
164
165 exclusiveScanGpu(counts.data() + firstNode, counts.data() + lastNode + 1, layout.data() + firstNode);
166 segmentMax(h, layout.data() + firstNode, numNodesSearch, d_radii + firstNode);
167 // SPH convention: interaction radius = 2 * h
168 scaleGpu(d_radii, d_radii + numLeafNodes, 2.0f);
169
170 fillGpu(d_flags, d_flags + numLeafNodes, 0);
171 findHalosGpu(prefixes, childOffsets, internalToLeaf, leaves, d_radii, box, firstNode, lastNode, d_flags);
172 memcpyD2H(d_flags, numLeafNodes, haloFlags_.data());
173
174 reallocateDevice(scratch, origSize, 1.0);
175 }
176 else
177 {
178 std::exclusive_scan(counts.begin() + firstNode, counts.begin() + lastNode + 1, layout.begin(), 0);
179 std::vector<float> haloRadii(counts.size(), 0.0f);
180#pragma omp parallel for schedule(static)
181 for (TreeNodeIndex i = 0; i < numNodesSearch; ++i)
182 {
183 if (layout[i + 1] > layout[i])
184 {
185 // Note factor 2 due to SPH convention: interaction radius = 2 * h
186 haloRadii[i + firstNode] = *std::max_element(h + layout[i], h + layout[i + 1]) * 2;
187 }
188 }
189 std::fill(begin(haloFlags_), end(haloFlags_), 0);
190 findHalos(prefixes, childOffsets, internalToLeaf, leaves, haloRadii.data(), box, firstNode, lastNode,
191 haloFlags_.data());
192 }
193 }

208 void computeLayout(gsl::span<const KeyType> leaves,
209 gsl::span<const unsigned> counts,
210 gsl::span<const TreeIndexPair> assignment,
211 gsl::span<const int> peers,
212 gsl::span<LocalIndex> layout)
213 {
214 computeNodeLayout(counts, haloFlags_, assignment[myRank_].start(), assignment[myRank_].end(), layout);
215 auto newParticleStart = layout[assignment[myRank_].start()];
216 auto newParticleEnd = layout[assignment[myRank_].end()];
217
218 outgoingHaloIndices_ = exchangeRequestKeys<KeyType>(leaves, haloFlags_, assignment, peers, layout);
219
220 detail::checkHalos(myRank_, assignment, haloFlags_);
221 detail::checkIndices(outgoingHaloIndices_, newParticleStart, newParticleEnd, layout.back());
222
223 incomingHaloIndices_ = computeHaloReceiveList(layout, haloFlags_, assignment, peers);
224 }

233 template<class Scratch1, class Scratch2, class... Vectors>
234 void exchangeHalos(std::tuple<Vectors&...> arrays, Scratch1& sendBuffer, Scratch2& receiveBuffer) const
235 {
236 if constexpr (HaveGpu<Accelerator>{})
237 {
238 static_assert(IsDeviceVector<Scratch1>{} && IsDeviceVector<Scratch2>{});
239 std::apply(
240 [this, &sendBuffer, &receiveBuffer](auto&... arrays)
241 {
242 haloExchangeGpu(haloEpoch_++, incomingHaloIndices_, outgoingHaloIndices_, sendBuffer, receiveBuffer,
243 rawPtr(arrays)...);
244 },
245 arrays);
246 }
247 else
248 {
249 std::apply([this](auto&... arrays)
250 { haloexchange(haloEpoch_++, incomingHaloIndices_, outgoingHaloIndices_, rawPtr(arrays)...); },
251 arrays);
252 }
253 }
